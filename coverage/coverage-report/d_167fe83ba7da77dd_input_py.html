25@deprecated(since='2.0.0', update_to='paddle.nn.functional.one_hot') 

26def one_hot(input, depth, allow_out_of_range=False): 

27 """ 

28 :alias_main: paddle.nn.functional.one_hot 

29 :alias: paddle.nn.functional.one_hot,paddle.nn.functional.common.one_hot 

30 :old_api: paddle.fluid.one_hot 

31 

32 The operator converts each id in the input to an one-hot vector with a 

33 depth length. The value in the vector dimension corresponding to the id 

34 is 1, and the value in the remaining dimension is 0. 

35 

36 The shape of output Tensor or LoDTensor is generated by appending depth dimension 

37 behind the last dimension of the input shape. 

38 

39 .. code-block:: text 

40 

41 Example 1 (allow_out_of_range=False): 

42 

43 input: 

44 X.shape = [4] 

45 X.data = [1, 1, 3, 0] 

46 depth = 4 

47 

48 output: 

49 Out.shape = [4, 4] 

50 Out.data = [[0., 1., 0., 0.], 

51 [0., 1., 0., 0.], 

52 [0., 0., 0., 1.], 

53 [1., 0., 0., 0.]] 

54 

55 Example 2 (allow_out_of_range=True): 

56 

57 input: 

58 X.shape = [4] 

59 X.data = [1, 1, 5, 0] 

60 depth = 4 

61 allow_out_of_range = True 

62 

63 output: 

64 Out.shape = [4, 4] 

65 Out.data = [[0., 1., 0., 0.], 

66 [0., 1., 0., 0.],  

67 [0., 0., 0., 0.], # This id is 5, which goes beyond depth, so set it all-zeros data. 

68 [1., 0., 0., 0.]] 

69 

70 Example 3 (allow_out_of_range=False): 

71 

72 input: 

73 X.shape = [4] 

74 X.data = [1, 1, 5, 0] 

75 depth = 4 

76 allow_out_of_range = False 

77 

78 output: Throw an exception for Illegal value 

79 The second dimension in X is 5, which is greater than depth.  

80 Allow_out_of_range =False means that does not allow the word id to exceed depth, 

81 so it throws an exception. 

82 

83 

84 Args: 

85 input(Variable): Tensor or LoDTensor with shape :math:`[N_1, N_2, ..., N_k]` , 

86 which contains at least one dimension. The data type is int32 or int64. 

87 depth(int): An integer defining the depth of the one hot dimension. If input  

88 is word id, depth is generally the dictionary size. 

89 allow_out_of_range(bool): A bool value indicating whether the input 

90 indices could be out of range :math:`[0, depth)` . When input indices are 

91 out of range, exceptions :code:`Illegal value` is raised if :attr:`allow_out_of_range` 

92 is False, or zero-filling representations is created if it is set True. 

93 Default: False. 

94 

95 Returns: 

96 Variable: The one-hot representations of input. A Tensor or LoDTensor with type float32. 

97 

98 Examples: 

99 .. code-block:: python 

100 

101 import paddle 

102 import paddle.fluid as fluid 

103 paddle.enable_static() 

104 

105 # Correspond to the first example above, where label.shape is 4 and one_hot_label.shape is [4, 4]. 

106 label = fluid.data(name="label", shape=[4], dtype="int64") 

107 one_hot_label = fluid.one_hot(input=label, depth=4) 

108 """ 

109 check_variable_and_dtype(input, 'input', ['int32', 'int64'], 'one_hot_v2') 

110 helper = LayerHelper("one_hot_v2", **locals()) 

111 

112 one_hot_out = helper.create_variable_for_type_inference(dtype='float32') 

113 

114 if _non_static_mode(): 

115 inputs = {'X': input} 

116 attrs = {'depth': depth, 'allow_out_of_range': allow_out_of_range} 

117 else: 

118 if not isinstance(depth, Variable): 

119 # user attribute 

120 inputs = {'X': input} 

121 attrs = {'depth': depth, 'allow_out_of_range': allow_out_of_range} 

122 else: 

123 depth.stop_gradient = True 

124 inputs = {'X': input, 'depth_tensor': depth} 

125 attrs = {'allow_out_of_range': allow_out_of_range} 

126 helper.append_op(type="one_hot_v2", 

127 inputs=inputs, 

128 attrs=attrs, 

129 outputs={'Out': one_hot_out}, 

130 stop_gradient=True) 

131 return one_hot_out

135@deprecated(since='2.0.0', update_to='paddle.nn.functional.embedding') 

136def embedding(input, 

137 size, 

138 is_sparse=False, 

139 is_distributed=False, 

140 padding_idx=None, 

141 param_attr=None, 

142 dtype='float32'): 

143 r""" 

144 :api_attr: Static Graph 

145 

146 The operator is used to lookup embeddings vector of ids provided by :attr:`input` .  

147 It automatically constructs a 2D embedding matrix based on the 

148 input :attr:`size` (vocab_size, emb_size) and :attr:`dtype` . 

149 

150 The shape of output Tensor is generated by appending an emb_size dimension to the 

151 last dimension of the input Tensor shape. 

152 

153 **Note:** The id in :attr:`input` must satisfy :math:`0 =< id < size[0]` ,  

154 otherwise the program will throw an exception and exit. 

155 

156 .. code-block:: text 

157 

158 Case 1: 

159 

160 input is a Tensor. padding_idx = -1 

161 input.data = [[1, 3], [2, 4], [4, 127]] 

162 input.shape = [3, 2] 

163 Given size = [128, 16] 

164 output is a Tensor: 

165 out.shape = [3, 2, 16] 

166 out.data = [[[0.129435295, 0.244512452, ..., 0.436322452], 

167 [0.345421456, 0.524563927, ..., 0.144534654]], 

168 

169 [[0.345249859, 0.124939536, ..., 0.194353745], 

170 [0.945345345, 0.435394634, ..., 0.435345365]], 

171  

172 [[0.945345345, 0.435394634, ..., 0.435345365], 

173 [0.0, 0.0, ..., 0.0 ]]] # padding data 

174 The input padding_idx is less than 0, it is automatically converted to padding_idx = -1 + 128 = 127 

175 It will pad all-zero data when ids is 127. 

176  

177 Case 2: 

178 

179 input is a LoDTensor with 1-level LoD. padding_idx = 0 

180 input.lod = [[2, 3]] 

181 input.data = [[1], [3], [2], [4], [0]] 

182 input.shape = [5, 1] 

183 Given size = [128, 16] 

184 output is a LoDTensor: 

185 out.lod = [[2, 3]] 

186 out.shape = [5, 1, 16] 

187 out.data = [[[0.129435295, 0.244512452, ..., 0.436322452]], 

188 [[0.345421456, 0.524563927, ..., 0.144534654]], 

189 [[0.345249859, 0.124939536, ..., 0.194353745]], 

190 [[0.945345345, 0.435394634, ..., 0.435345365]], 

191 [[0.0, 0.0, ..., 0.0 ]]] # padding data 

192 It will pad all-zero data when ids is 0. 

193 

194 

195 Args: 

196 input(Variable): A Tensor or LoDTensor with type int64, which contains the id information. 

197 The value of the input id should satisfy :math:`0<= id < size[0]` . 

198 size(tuple|list): The shape of lookup table parameter. It should have two elements which 

199 indicates the size of the dictionary of embeddings and the size of each embedding vector respectively. 

200 is_sparse(bool): The flag indicating whether to use sparse update. This parameter only 

201 affects the performance of the backwards gradient update. It is recommended to set  

202 True because sparse update is faster. But some optimizer does not support sparse update 

203 In these case, is_sparse must be False. Default: False. 

204 is_distributed(bool): Whether to store the embedding matrix in a distributed manner. Only used 

205 in multi-machine distributed CPU training. Default: False. 

206 padding_idx(int|long|None): padding_idx needs to be in the interval [-vocab_size, vocab_size).  

207 If :math:`padding\_idx < 0`, the :math:`padding\_idx` will automatically be converted 

208 to :math:`vocab\_size + padding\_idx` . It will output all-zero padding data whenever lookup 

209 encounters :math:`padding\_idx` in id. And the padding data will not be updated while training. 

210 If set None, it makes no effect to output. Default: None. 

211 param_attr(ParamAttr): To specify the weight parameter property. Default: None, which means the 

212 default weight parameter property is used. In addition, 

213 user-defined or pre-trained word vectors can be loaded with the :attr:`param_attr` parameter.  

214 The local word vector needs to be transformed into numpy format, and the shape of local word 

215 vector should be consistent with :attr:`size` . 

216 dtype(str): It refers to the data type of output Tensor. 

217 It must be float32 or float64. Default: float32. 

218 

219 Returns: 

220 Variable: Embedding Tensor or LoDTensor mapped by input. The data type is the same as :attr:`dtype` . 

221 

222 Static Examples: 

223 .. code-block:: python 

224 

225 import paddle 

226 import numpy as np 

227 paddle.enable_static() 

228  

229 x = paddle.static.data(name="x", shape = [2, 4], dtype=np.int64) 

230 embedding = paddle.nn.Embedding(10, 3, 

231 weight_attr=paddle.nn.initializer.Constant(value=1.0)) 

232 adam = paddle.optimizer.SGD(parameters=[embedding.weight], learning_rate=0.01) 

233 output = embedding(x) 

234 m_output=paddle.mean(output) 

235  

236 adam.minimize(m_output) 

237  

238 place = paddle.CPUPlace() 

239 exe = paddle.static.Executor(place) 

240 exe.run(paddle.static.default_startup_program()) 

241  

242 x = np.array([[7, 2, 4, 5],[4, 3, 2, 9]], dtype=np.int64) 

243  

244 # x is a Numpy. 

245 # x.data = [[7, 2, 4, 5], [4, 3, 2, 9]] 

246 # x.shape = [2, 4] 

247  

248 out, = exe.run(paddle.static.default_main_program(), feed={'x':x}, fetch_list=[output]) 

249  

250 # out is a Numpy. 

251 # out.data = [[1., 1., 1.], 

252 # [1., 1., 1.], 

253 # [1., 1., 1.], 

254 # [1., 1., 1.]],

256 # [[1., 1., 1.], 

257 # [1., 1., 1.],

304 # out.data: [2 2 2 2 2 2 2 2 2] 

305 out = embedding(x) 

306  

307 out.backward() 

308 adam.step() 

309 

310 """ 

311 

312 helper = LayerHelper('embedding', **locals()) 

313 check_variable_and_dtype(input, 'input', ['int64'], 'fluid.embedding') 

314 check_dtype(dtype, 'dtype', ['float16', 'float32', 'float64', 'uint16'], 

315 'fluid.embedding') 

316 remote_prefetch = is_sparse and (not is_distributed) 

317 if remote_prefetch: 317 ↛ 318line 317 didn't jump to line 318, because the condition on line 317 was never true

318 assert is_sparse is True and is_distributed is False 

319 w = helper.create_parameter(attr=helper.param_attr, 

320 shape=size, 

321 dtype=dtype, 

322 is_bias=False) 

323 tmp = helper.create_variable_for_type_inference(dtype) 

324 padding_idx = -1 if padding_idx is None else padding_idx if padding_idx >= 0 else ( 

325 size[0] + padding_idx) 

326 helper.append_op(type='lookup_table_v2', 

327 inputs={ 

328 'Ids': input, 

329 'W': w 

330 }, 

331 outputs={'Out': tmp}, 

332 attrs={ 

333 'is_sparse': is_sparse, 

334 'is_distributed': is_distributed, 

335 'remote_prefetch': remote_prefetch, 

336 'padding_idx': padding_idx 

337 }) 

338 return tmp
