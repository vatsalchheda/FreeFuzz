52def apply_optimization(transformers): 

53 """ 

54 Judge wheter to apply optimized transformation, such as BreakTransformOptimizer. 

55 And not all optimized transformations are applied by default. It's controlled by 

56 'export FLAGS_optim_transformation=1' 

57 """ 

58 flag = str( 

59 os.environ.get('FLAGS_optim_transformation')) in ['1', 'True', 'true'] 

60 if flag: 

61 transformers.insert(3, BreakTransformOptimizer)

64class DygraphToStaticAst(BaseTransformer): 

65 """ 

66 Main class to transform Dygraph to Static Graph 

67 """ 

68 

69 def __init__(self): 

70 self.translator_logger = logging_utils.TranslatorLogger() 

71 

72 def get_static_ast(self, root): 

73 # save root for some analysis may need global AST 

74 self.root = root 

75 self.static_analysis_visitor = StaticAnalysisVisitor(root) 

76 self.static_analysis_root = self.static_analysis_visitor.get_node_wrapper_root( 

77 ) 

78 self.decorate_func_name = None 

79 self.transfer_from_node_type(self.static_analysis_root) 

80 return self.static_analysis_root 

81 

82 def _apply(self, transformer, node_wrapper, log_level): 

83 transformer(node_wrapper).transform() 

84 self.translator_logger.log_transformed_code(log_level, self.root, 

85 transformer.__name__) 

86 

87 def transfer_from_node_type(self, node_wrapper): 

88 self.translator_logger.log( 

89 1, "Source code: \n{}".format(ast_to_source_code(self.root))) 

90 # Generic transformation 

91 self.visit(node_wrapper.node) 

92 

93 transformers = [ 

94 EarlyReturnTransformer, 

95 BasicApiTransformer, # Basic Api 

96 TensorShapeTransformer, # Tensor.shape -> layers.shape(Tensor) 

97 #ListTransformer, # List used in control flow 

98 BreakContinueTransformer, # break/continue in loops 

99 ReturnTransformer, # return in functions 

100 LogicalTransformer, # logical and/or/not 

101 CreateVariableTransformer, # create undefined var for if / while / for 

102 LoopTransformer, # for/while -> while_op 

103 IfElseTransformer, # if/else -> cond_op 

104 AssertTransformer, # assert statement 

105 PrintTransformer, # print statement 

106 CallTransformer, # transform call recursively 

107 CastTransformer, # type casting statement 

108 #GradTransformer, # transform paddle.grad to paddle.gradients 

109 DecoratorTransformer, # transform decorators to function call 

110 TypeHintTransformer, # remove all typehint in gast.Name 

111 ] 

112 

113 apply_optimization(transformers) 

114 

115 for index, transformer in enumerate(transformers): 

116 self._apply(transformer, node_wrapper, log_level=index + 1) 

117 

118 self.translator_logger.log_transformed_code( 

119 logging_utils.LOG_AllTransformer, self.root, "All Transformers")

121 def visit_FunctionDef(self, node): 

122 if self.decorate_func_name is None: 

123 self.decorate_func_name = node.name 

124 

125 self.generic_visit(node) 

126 return node
