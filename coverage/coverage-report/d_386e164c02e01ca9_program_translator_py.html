57class FunctionCache(object): 

58 """ 

59 Caches the transformed functions to avoid redundant conversions of the same function. 

60 """

62 def __init__(self): 

63 # Caches the converted static functions. {dygraph_func: static_func} 

64 self._converted_static_func_caches = weakref.WeakKeyDictionary() 

65 # Caches the converted ast node for same source code. {source_code: ast_root} 

66 self._code_to_ast_caches = dict() 

67 self._dygraph_to_static = DygraphToStaticAst()

69 def convert_with_cache(self, func): 

70 """ 

71 Returns the cached static function or converts it when first encounters the function. 

72 """ 

73 # If hit cache, return it directly. 

74 static_func = self._converted_static_func_caches.get(func, None) 

75 

76 if static_func is None: 

77 static_func = self._convert(func) 

78 self._converted_static_func_caches[func] = static_func 

79 

80 return static_func

82 def _convert(self, func): 

83 """ 

84 Converts dygraph function into static function. For two functions with same dedent code, 

85 the second function will reuse the transformed ast node of previous one. 

86 

87 For example: 

88 # A.py 

89 def foo(x, y): 

90 z = x + y 

91 return z 

92 

93 # B.py 

94 def foo(x, y): 

95 z = x + y 

96 return z 

97 

98 If the conversion of A.foo happens after B.foo, it will reuse the transformed ast node of B.foo 

99 to speed up the conversion. 

100 """ 

101 # Note: In Python2, it will raise OSError when inspect function 

102 # with decorator directly and function.__wrapped__ holds the actual function. 

103 func = unwrap(func) 

104 source_code = func_to_source_code(func) 

105 

106 # TODO(liym27): 

107 # Consider this case: source_code in self._code_to_ast_caches, 

108 # but actually they are methods in different classes. 

109 # Maybe use (__class__, source_code) as key 

110 if source_code in self._code_to_ast_caches: 

111 root_wrapper = self._code_to_ast_caches[source_code] 

112 else: 

113 root = gast.parse(source_code) 

114 root = attach_origin_info(root, func) 

115 root_wrapper = self._dygraph_to_static.get_static_ast(root) 

116 self._code_to_ast_caches[source_code] = root_wrapper 

117 

118 # Get static function from AST 

119 static_func, file_name = ast_to_func(root_wrapper.node, func) 

120 

121 create_and_update_origin_info_map(root_wrapper.node, static_func) 

122 return static_func 

123 

124 def exist(self, func): 

125 return func in self._converted_static_func_caches

309 def __get__(self, instance, owner): 

310 """ 

311 Overrides this method to parse the class instance and call bound method correctly. 

312 

313 For example: 

314  

315 ''' 

316 class Net(Layer): 

317 def __init__(self): 

318 pass 

319  

320 @paddle.jit.to_static 

321 def forward(self, x, y): 

322 return x + y 

323 

324 net = Net() 

325 out = net(x, y) 

326 ''' 

327  

328 In above case, `net(x, y)` will call `net.forward(x, y)` firstly that is a bound method 

329 of `Net` instance. After decorated by `@paddle.jit.to_static`, it will firstly to call `__get__` 

330 to parse the class instance correctly instead of the `StaticFunction` instance. 

331 """ 

332 if instance not in self._descriptor_cache: 

333 if instance is None: 

334 return self 

335 # Note(Aurelius84): To construct new instance of StaticFunction when we 

336 # first encouter the bound function of layer and cache it. 

337 new_static_layer = self._clone() 

338 new_static_layer._class_instance = instance 

339 self._descriptor_cache[instance] = new_static_layer 

340 

341 return self._descriptor_cache[instance]

347 def __call__(self, *args, **kwargs): 

348 """ 

349 Supports to call the returned instance with input `args` and `kwargs` directly. 

350 

351 Args: 

352 *args(tuple): tuple of all input arguments from original decorated function. 

353 **kwargs(dict): dict of all input keyward arguments from original decorated function.  

354 

355 Return: 

356 Outputs of decorated function. 

357 """ 

358 if self._property: 

359 return self._call_dygraph_function(*args, **kwargs) 

360 

361 # 1. call dygraph function directly if not enable `declarative` 

362 if not self._program_trans.enable_to_static: 

363 # NOTE(liym27): 

364 # Here calls `warnings.warn` but not `logging_utils.warn` because by default warnings.warn(message) 

365 # will show up **only once**. StaticFunction.__call__ will run many times, it is appropriate to 

366 # display this warning message only once. 

367 logging_utils.warn( 

368 "The decorator '@paddle.jit.to_static' does NOT work when setting ProgramTranslator.enable to False. " 

369 "We will just return dygraph output. If you would like to get static graph output, please call API " 

370 "ProgramTranslator.enable(True)") 

371 return self._call_dygraph_function(*args, **kwargs) 

372 

373 if not _non_static_mode(): 

374 raise RuntimeError( 

375 "Failed to run the callable object {} decorated by '@paddle.jit.to_static', " 

376 "because it is NOT in dynamic mode. Please disable the static mode to enter dynamic mode with the " 

377 "following API: paddle.disable_static().".format( 

378 self.dygraph_function)) 

379 

380 # 2. trace ops from dygraph layers and cache the generated program. 

381 args, kwargs = self._function_spec.unified_args_and_kwargs(args, kwargs) 

382 

383 try: 

384 concrete_program, partial_program_layer = self.get_concrete_program( 

385 *args, **kwargs, is_train=self._is_train_mode()) 

386 # 3. synchronize self.training attribute. 

387 if isinstance(self._class_instance, layers.Layer): 

388 partial_program_layer.training = self._class_instance.training 

389 else: 

390 partial_program_layer.training = self._training 

391 

392 partial_program_layer._cuda_graph_capture_mode = self._cuda_graph_capture_mode 

393 partial_program_layer._cuda_graph_pool_id = self._cuda_graph_pool_id 

394 

395 # 4. return outputs. 

396 try: 

397 return partial_program_layer(args) 

398 except Exception as e: 

399 if not hasattr(e, error.ERROR_DATA): 

400 # runtime error 

401 error.attach_error_data(e, in_runtime=True) 

402 raise 

403 except Exception as e: 

404 error_data = getattr(e, error.ERROR_DATA, None) 

405 if error_data: 

406 error_data.raise_new_exception() 

407 else: 

408 logging_utils.warn( 

409 "Please file an issue at 'https://github.com/PaddlePaddle/Paddle/issues'" 

410 " if you can't handle this {} yourself.".format(type(e))) 

411 raise e

451 def get_concrete_program(self, *args, **kwargs): 

452 """ 

453 Returns traced concrete program and inner executable partial layer. 

454 

455 Args: 

456 *args(tuple): input arguments values or InputSpec 

457 **kwargs(dict) : input kwargs values. 

458 

459 Returns: 

460 Traced ConcreteProgram and executable translated Layer. 

461 """ 

462 self._raise_when_property() 

463 

464 with_hook = kwargs.get("with_hook", False) 

465 is_train = kwargs.get("is_train", True) 

466 if "is_train" in kwargs: kwargs.pop("is_train") 

467 if "with_hook" in kwargs: kwargs.pop("with_hook") 

468 # 1. unify args/kwargs and replace Tensor with InputSpec 

469 if len(args) != len(self._function_spec.args_name): 

470 args, kwargs = self._function_spec.unified_args_and_kwargs( 

471 args, kwargs) 

472 input_args_with_spec, input_kwargs_with_spec = self._function_spec.args_to_input_spec( 

473 args, kwargs) 

474 

475 # 2. generate cache key 

476 cache_key = CacheKey(self._function_spec, 

477 input_args_with_spec, 

478 input_kwargs_with_spec, 

479 self._class_instance, 

480 **self._kwargs, 

481 with_hook=with_hook, 

482 is_train=is_train) 

483 

484 # 3. check whether hit the cache or build a new program for the input arguments 

485 concrete_program, partial_program_layer = self._program_cache[cache_key] 

486 return concrete_program, partial_program_layer

539 def concrete_program_specify_input_spec(self, 

540 input_spec=None, 

541 with_hook=False): 

542 """ 

543 Returns recent ConcreteProgram instance of decorated function while 

544 specifying input_spec. If the self._function_spec already has 

545 input_spec, it will check the compatibility of input input_spec and 

546 the self._function_spec.input_spec. If input input_spec=None, then 

547 this method uses self._function_spec.input_spec 

548 

549 args: 

550 input_spec (list[InputSpec], optional): Describes the input of 

551 the translate function. 

552 """ 

553 self._raise_when_property() 

554 # if specific the `input_spec`, the length of program_cache will always 1, 

555 # else, return the last one. 

556 cached_program_len = len(self._program_cache) 

557 # If specific `input_spec`, apply convertion from dygraph layers into static Program. 

558 if cached_program_len == 0: 

559 desired_input_spec = input_spec 

560 if self._function_spec.input_spec is not None: 

561 if input_spec is not None and not input_specs_compatible( 

562 flatten(input_spec), 

563 flatten(self._function_spec.input_spec)): 

564 raise ValueError( 

565 "The `input_spec`: {} used to construct concrete_program is conflict with the `input_spec`: {} in `@paddle.jit.to_static`" 

566 .format(input_spec, self._function_spec.input_spec)) 

567 # NOTE(chenweihang): we should always translated program based on the `input_spec` 

568 # decorated on forward if it is valid 

569 desired_input_spec = self._function_spec.input_spec 

570 if input_spec is not None: 

571 logging_utils.warn( 

572 "\n\nYou have specified `input_spec` both in function definition (higher priority) and `paddle.jit.save` (will be ignored.)\n\n\t Using: {}\n\n\t Ignore: {}\n" 

573 .format(desired_input_spec, input_spec)) 

574 

575 has_input_spec = (desired_input_spec is not None) 

576 if has_input_spec: 

577 concrete_program, _ = self.get_concrete_program( 

578 *desired_input_spec, 

579 with_hook=with_hook, 

580 is_train=self._is_train_mode()) 

581 return concrete_program 

582 else: 

583 raise ValueError( 

584 "No valid transformed program for {}.\n\t Please specific `input_spec` in `@paddle.jit.to_static` or feed input tensor to call the decorated function at once.\n" 

585 .format(self._function_spec)) 

586 elif with_hook: 

587 cache_key = self._program_cache._recent_cache_key 

588 cache_key.kwargs["with_hook"] = True 

589 concrete_program, _ = self._program_cache[cache_key] 

590 return concrete_program 

591 

592 # If more than one programs have been cached, return the recent converted program by default. 

593 elif cached_program_len > 1: 

594 logging_utils.warn( 

595 "Current {} has more than one cached programs: {}, the last traced progam will be return by default." 

596 .format(self._function_spec, cached_program_len)) 

597 

598 cache_key, (concrete_program, 

599 partial_layer) = self._program_cache.last() 

600 return concrete_program

634 for name, func in class_instance._original_funcs.items(): 

635 setattr(class_instance, name, func.__get__(class_instance))

638 rollback_impl(sublayer)

641 return self._dygraph_function

644 func_name = self._dygraph_function.__name__ 

645 assert func_name in self._class_instance._original_funcs, "Not Found function '{}' in class '{}'.".format( 

646 func_name, self._class_instance.__name__) 

647 func = self._class_instance._original_funcs[func_name] 

648 setattr(self._class_instance, func_name, 

649 func.__get__(self._class_instance)) 

650

652 rollback_impl(sublayer) 

653 

654 return getattr(self._class_instance, func_name)

760class HookHelper(object): 

761 """ 

762 Only For converting pre/post hooks operation in outermost layer while jit.save. 

763 Because hooks in sublayer have been processed automatically. 

764 """ 

765 

766 def __init__(self, func, class_instance, with_hook=False): 

767 self.func = func 

768 self.class_instance = class_instance 

769 self.with_hook = with_hook 

770 self.need_apply_hook = with_hook and isinstance( 

771 self.class_instance, layers.Layer) and getattr( 

772 func, "__name__") == "forward" 

773 

774 def apply_pre_hooks(self, inputs): 

775 """ 

776 Apply _forward_pre_hooks from outermost layer 

777 """ 

778 if not self.need_apply_hook: return inputs 

779 

780 inputs = inputs[1:] 

781 for forward_pre_hook in self.class_instance._forward_pre_hooks.values(): 

782 hook_result = forward_pre_hook(self.class_instance, inputs) 

783 if hook_result is not None: 

784 if not isinstance(hook_result, tuple): 

785 hook_result = (hook_result, ) 

786 inputs = hook_result 

787 

788 return [self.class_instance] + list(inputs) 

789 

790 def apply_post_hooks(self, inputs, outputs): 

791 """ 

792 Apply _forward_post_hooks from outermost layer 

793 """ 

794 if not self.need_apply_hook: return outputs 

795 

796 inputs = inputs[1:] 

797 for forward_post_hook in self.class_instance._forward_post_hooks.values( 

798 ): 

799 hook_result = forward_post_hook(self.class_instance, inputs, 

800 outputs) 

801 if hook_result is not None: 

802 outputs = hook_result 

803 

804 inputs.insert(0, self.class_instance) 

805 return outputs

833 def from_func_spec(func_spec, input_spec, input_kwargs_spec, class_instance, 

834 **kwargs): 

835 """ 

836 Builds the main_program with specialized inputs and returns outputs 

837 of program as fetch_list. 

838 

839 Args: 

840 func_spec(FunctionSpec): A FunctionSpec instance for decorated function. 

841 input_spec(list[InputSpec]):  

842 """ 

843 # verify the instance is initialized in imperative mode. 

844 _verify_init_in_dynamic_mode(class_instance) 

845 

846 # Transforms dygraph function into static function and caches it. 

847 dygraph_function = func_spec.dygraph_function 

848 static_func = convert_to_static(dygraph_function) 

849 # apply pre\post hook for outermost layer 

850 hook_helper = HookHelper(dygraph_function, class_instance, 

851 kwargs.get("with_hook", False)) 

852 

853 main_program, startup_program = framework.Program(), framework.Program() 

854 # Note: The random seed should be synchronized into cached program 

855 # if set in `fluid.dygraph_guard` because some ops rely on it, such as 

856 # `fluid.layers.dropout`. 

857 main_program.random_seed = framework.default_main_program().random_seed 

858 startup_program.random_seed = framework.default_startup_program( 

859 ).random_seed 

860 

861 from paddle.fluid.dygraph.base import _switch_declarative_mode_guard_ 

862 with framework.program_guard(main_program, startup_program): 

863 with _switch_declarative_mode_guard_(is_declarative=True): 

864 # 1. Adds `fluid.data` layers for input if needed 

865 static_inputs = func_spec.to_static_inputs_with_spec( 

866 input_spec, main_program) 

867 _kwargs = func_spec.to_static_inputs_with_spec( 

868 input_kwargs_spec, main_program) 

869 if class_instance: 

870 static_inputs = tuple([class_instance] + 

871 list(static_inputs)) 

872 

873 # 2. Gets all ParamBases and buffered VarBases in the function 

874 all_parameters_and_buffers = _extract_indeed_params_buffers( 

875 class_instance) 

876 

877 # 3. Builds program only once and returns the output Variables. 

878 with param_guard(get_parameters( 

879 class_instance, 

880 False)), param_guard(get_buffers(class_instance, 

881 False)): 

882 try: 

883 # only for jit.save, do nothing while train and eval process 

884 inputs = hook_helper.apply_pre_hooks(static_inputs) 

885 if _kwargs: 

886 outputs = static_func(*inputs, **_kwargs) 

887 else: 

888 outputs = static_func(*inputs) 

889 outputs = hook_helper.apply_post_hooks(inputs, outputs) 

890 except BaseException as e: 

891 # NOTE: If e is raised in compile time, e should be attached to ERROR_DATA here. 

892 error.attach_error_data(e) 

893 error_data = getattr(e, error.ERROR_DATA, None) 

894 if error_data: 

895 error_data.raise_new_exception() 

896 raise 

897 

898 if outputs is not None: 

899 need_wrap_into_list = not isinstance( 

900 outputs, (tuple, list)) or len(outputs) == 1 

901 if need_wrap_into_list: 

902 outputs = [outputs] 

903 

904 main_program = update_op_callstack_with_origin_info(main_program) 

905 

906 return ConcreteProgram(inputs=static_inputs, 

907 outputs=outputs, 

908 parameters=all_parameters_and_buffers, 

909 function=dygraph_function, 

910 main_program=main_program, 

911 startup_program=startup_program, 

912 **kwargs)

928 Wrapper class for the program functions defined by dygraph function.

930 

931 def __init__(self): 

932 # {hash_id : (concrete_program, partial_layer)} 

933 self._caches = collections.OrderedDict() 

934 # trace mostly recent used program 

935 self._recent_key = None 

936 self._recent_cache_key = None 

937 

938 def _build_once(self, cache_key): 

939 concrete_program = ConcreteProgram.from_func_spec( 

940 func_spec=cache_key.function_spec, 

941 input_spec=cache_key.input_args_with_spec, 

942 input_kwargs_spec=cache_key.input_kwargs_with_spec, 

943 class_instance=cache_key.class_instance, 

944 **cache_key.kwargs) 

945 return concrete_program, partial_program_from(concrete_program)

947 def __getitem__(self, item): 

948 if not isinstance(item, CacheKey): 

949 raise ValueError('type(item) should be CacheKey, but received %s' % 

950 type_name(item)) 

951 item_id = hash(item) 

952 self._recent_cache_key = item 

953 self._recent_key = item_id 

954 if item_id not in self._caches: 

955 self._caches[item_id] = self._build_once(item) 

956 # Note: raise warnings if number of traced program is more than `max_tracing_count` 

957 current_tracing_count = len(self._caches) 

958 if current_tracing_count > MAX_TRACED_PROGRAM_COUNT: 

959 logging_utils.warn( 

960 "Current traced program number: {} > `max_tracing_count`:{}. Too much cached programs will bring expensive overhead. " 

961 "The reason may be: (1) passing tensors with different shapes, (2) passing python objects instead of tensors." 

962 .format(current_tracing_count, MAX_TRACED_PROGRAM_COUNT)) 

963 

964 return self._caches[item_id] 

965 

966 def get_program(self, item): 

967 if not isinstance(item, CacheKey): 

968 raise ValueError( 

969 "Input item's type should be FunctionSpec, but received %s" % 

970 type_name(item)) 

971 item_id = hash(item) 

972 if item_id not in self._caches: 

973 raise RuntimeError( 

974 "Failed to find program for input item, please decorate input function by `@paddle.jit.to_static`." 

975 ) 

976 return self._caches[item_id]

979 assert len(

981 assert self._recent_key is not None
