35def _parse_load_config(configs): 

36 supported_configs = ['model_filename', 'params_filename', 'keep_name_table'] 

37 

38 # input check 

39 for key in configs: 

40 if key not in supported_configs: 

41 raise ValueError( 

42 "The additional config (%s) of `paddle.fluid.load_dygraph` is not supported." 

43 % (key)) 

44 

45 # construct inner config 

46 inner_config = _SaveLoadConfig() 

47 inner_config.model_filename = configs.get('model_filename', None) 

48 inner_config.params_filename = configs.get('params_filename', None) 

49 inner_config.keep_name_table = configs.get('keep_name_table', None) 

50 

51 return inner_config

54@dygraph_only 

55def save_dygraph(state_dict, model_path): 

56 ''' 

57 :api_attr: imperative 

58 

59 Save Layer's state_dict to disk. This will generate a file with suffix ".pdparams" 

60  

61 The state_dict is get from Layers.state_dict function 

62  

63 Args: 

64 state_dict(dict) : The state dict to be saved. 

65 model_path(str) : the file prefix to save the state_dict. The format is "dirname/file_prefix". If file_prefix is empty str. A exception will be raised 

66 

67 Returns: 

68 None 

69 

70 Examples: 

71 .. code-block:: python 

72 

73 import paddle.fluid as fluid 

74 

75 with fluid.dygraph.guard(): 

76 emb = fluid.dygraph.Embedding([10, 10]) 

77 

78 state_dict = emb.state_dict() 

79 fluid.save_dygraph( state_dict, "paddle_dy") 

80 

81 adam = fluid.optimizer.Adam( learning_rate = fluid.layers.noam_decay( 100, 10000), 

82 parameter_list = emb.parameters() ) 

83 

84 state_dict = adam.state_dict() 

85 fluid.save_dygraph( state_dict, "paddle_dy") 

86 

87 ''' 

88 

89 base_name = os.path.basename(model_path) 

90 assert base_name != "", "The input model_path MUST be format of dirname/filename [dirname\\filename in Windows system], but received filename is empty string." 

91 

92 suffix = ".pdparams" 

93 assert len(state_dict) > 0, "state_dict is empty, no need to save" 

94 

95 param_num = 0 

96 for k, v in state_dict.items(): 

97 if isinstance(v, (ParamBase, EagerParamBase)): 

98 param_num += 1 

99 

100 if param_num == 0: 

101 suffix = ".pdopt" 

102 

103 model_dict = {} 

104 name_table = {} 

105 for k, v in state_dict.items(): 

106 if isinstance(v, (Variable, core.VarBase, core.eager.Tensor)): 

107 model_dict[k] = v.numpy() 

108 name_table[k] = v.name 

109 else: 

110 model_dict[k] = v 

111 model_dict["StructuredToParameterName@@"] = name_table 

112 

113 file_name = model_path + suffix 

114 dir_name = os.path.dirname(file_name) 

115 if dir_name and not os.path.exists(dir_name): 

116 os.makedirs(dir_name) 

117 

118 with open(file_name, 'wb') as f: 

119 pickle.dump(model_dict, f, protocol=2)

127def load_dygraph(model_path, **configs): 

128 ''' 

129 :api_attr: imperative 

130  

131 Load parameter state dict from disk. 

132 

133 .. note:: 

134 Due to some historical reasons, if you load ``state_dict`` from the saved  

135 result of `paddle.static.save_inference_model`, the structured variable name  

136 will cannot be restored. You need to set the argument `use_structured_name=False`  

137 when using `Layer.set_state_dict` later. 

138 

139 Args: 

140 model_path(str) : The file prefix store the state_dict.  

141 (The path should Not contain suffix '.pdparams')  

142 **configs (dict, optional): Other load configuration options for compatibility. We do not  

143 recommend using these configurations, if not necessary, DO NOT use them. Default None. 

144 The following options are currently supported: 

145 (1) model_filename (str): The inference model file name of the paddle 1.x ``save_inference_model``  

146 save format. Default file name is :code:`__model__` .  

147 (2) params_filename (str): The persistable variables file name of the paddle 1.x ``save_inference_model``  

148 save format. No default file name, save variables separately by default. 

149 

150 Returns: 

151 state_dict(dict) : the dict store the state_dict 

152 

153 Examples: 

154 .. code-block:: python 

155 

156 import paddle 

157 import paddle.fluid as fluid 

158 

159 paddle.disable_static() 

160 

161 emb = paddle.nn.Embedding(10, 10) 

162 

163 state_dict = emb.state_dict() 

164 fluid.save_dygraph(state_dict, "paddle_dy") 

165 

166 scheduler = paddle.optimizer.lr.NoamDecay(  

167 d_model=0.01, warmup_steps=100, verbose=True) 

168 adam = paddle.optimizer.Adam( 

169 learning_rate=scheduler, 

170 parameters=emb.parameters()) 

171 state_dict = adam.state_dict() 

172 fluid.save_dygraph(state_dict, "paddle_dy") 

173 

174 para_state_dict, opti_state_dict = fluid.load_dygraph("paddle_dy") 

175 ''' 

176 # deal with argument `model_path` 

177 model_prefix = model_path 

178 if model_prefix.endswith(".pdparams"): 

179 model_prefix = model_prefix[:-9] 

180 elif model_prefix.endswith(".pdopt"): 

181 model_prefix = model_prefix[:-6] 

182 

183 para_dict = None 

184 opti_dict = None 

185 params_file_path = model_prefix + ".pdparams" 

186 opti_file_path = model_prefix + ".pdopt" 

187 

188 # deal with argument `config` 

189 config = _parse_load_config(configs) 

190 

191 if os.path.exists(params_file_path) or os.path.exists(opti_file_path): 

192 # Load state dict by `save_dygraph` save format 

193 para_dict = {} 

194 if os.path.exists(params_file_path): 

195 with open(params_file_path, 'rb') as f: 

196 para_dict = pickle.load(f, encoding='latin1') 

197 

198 if not config.keep_name_table and "StructuredToParameterName@@" in para_dict: 

199 del para_dict["StructuredToParameterName@@"] 

200 

201 if os.path.exists(opti_file_path): 

202 with open(opti_file_path, 'rb') as f: 

203 opti_dict = pickle.load(f, encoding='latin1') 

204 else: 

205 # check model path 

206 if not os.path.isdir(model_prefix): 

207 raise ValueError("Model saved directory '%s' is not exists." % 

208 model_prefix) 

209 

210 # check whether model file exists 

211 if config.model_filename is None: 

212 model_filename = '__model__' 

213 else: 

214 model_filename = config.model_filename 

215 model_file_path = os.path.join(model_path, model_filename) 

216 

217 if os.path.exists(model_file_path): 

218 # Load state dict by `jit.save/io.save_inference_model` save format 

219 # NOTE(chenweihang): [ Compatibility of save_inference_model save format ] 

220 # The model saved by `save_inference_model` does not completely correspond to 

221 # the information required by the `state_dict` under the dygraph. 

222 # `save_inference_model` not save structured name, we need to remind 

223 # the user to configure the `use_structured_name` argument when `set_state_dict` 

224 # NOTE(chenweihang): `jit.save` doesn't save optimizer state 

225 

226 # 1. load program desc & construct _ProgramHolder 

227 programs = _construct_program_holders(model_path, 

228 config.model_filename) 

229 

230 # 2. load layer parameters & buffers 

231 # NOTE: using fluid.dygraph.guard() here will cause import error in py2 

232 with guard(): 

233 persistable_var_dict = _construct_params_and_buffers( 

234 model_prefix, 

235 programs, 

236 config.params_filename, 

237 append_suffix=False) 

238 

239 # 3. construct state_dict 

240 para_dict = dict() 

241 for var_name in persistable_var_dict: 

242 para_dict[var_name] = persistable_var_dict[var_name].numpy() 

243 

244 # if *.info exists, we can recover structured_name 

245 var_info_filename = str(config.params_filename) + ".info" 

246 var_info_path = os.path.join(model_prefix, var_info_filename) 

247 if os.path.exists(var_info_path): 

248 with open(var_info_path, 'rb') as f: 

249 extra_var_info = pickle.load(f) 

250 structured_para_dict = dict() 

251 for var_name in para_dict: 

252 structured_name = extra_var_info[var_name].get( 

253 'structured_name', None) 

254 assert structured_name is not None, "Cannot find saved variable (%s)'s structured name in saved model." % var_name 

255 structured_para_dict[structured_name] = para_dict[ 

256 var_name] 

257 para_dict = structured_para_dict 

258 else: 

259 # load state dict by `io.save_params/persistables` save format 

260 # TODO(chenweihang): [ Now only supports loading parameters separately ] 

261 # If users save all parameters as one file, the [ variable.name -> variable ] 

262 # mapping info will lost, so users need to give variable list, but users build 

263 # variable list in dygraph mode is difficult, we recommend users to use 

264 # paddle.static.load_program_state in this case 

265 

266 # Try to load all the files in the directory in VarBase format, 

267 # the file name is used as the name of VarBase 

268 load_var_list = [] 

269 

270 # 1. load file names 

271 var_name_list = [] 

272 for root, _, files in os.walk(model_path): 

273 for filename in files: 

274 file_path = os.path.join(root, filename) 

275 tmp_var_name = os.path.relpath(file_path, model_path) 

276 var_name = tmp_var_name.replace("\\", "/") 

277 var_name_list.append(var_name) 

278 

279 # 2. create and load VarBase 

280 with guard(): 

281 for name in var_name_list: 

282 new_var = _varbase_creator(name=name, persistable=True) 

283 _dygraph_tracer().trace_op( 

284 type='load', 

285 inputs={}, 

286 outputs={'Out': new_var}, 

287 attrs={'file_path': os.path.join(model_path, name)}) 

288 load_var_list.append(new_var) 

289 

290 # 3. construct state_dict 

291 para_dict = dict() 

292 for var in load_var_list: 

293 para_dict[var.name] = var.numpy() 

294 

295 return para_dict, opti_dict
