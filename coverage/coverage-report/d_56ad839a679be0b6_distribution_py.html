55class Distribution(object): 

56 """ 

57 The abstract base class for probability distributions. Functions are 

58 implemented in specific distributions. 

59 

60 Args: 

61 batch_shape(Sequence[int], optional): independent, not identically 

62 distributed draws, aka a "collection" or "bunch" of distributions. 

63 event_shape(Sequence[int], optional): the shape of a single 

64 draw from the distribution; it may be dependent across dimensions. 

65 For scalar distributions, the event shape is []. For n-dimension 

66 multivariate distribution, the event shape is [n]. 

67 """

161 def _validate_args(self, *args): 

162 """ 

163 Argument validation for distribution args 

164 Args: 

165 value (float, list, numpy.ndarray, Tensor) 

166 Raises 

167 ValueError: if one argument is Tensor, all arguments should be Tensor 

168 """ 

169 is_variable = False 

170 is_number = False 

171 for arg in args: 

172 if isinstance(arg, tensor.Variable): 

173 is_variable = True 

174 else: 

175 is_number = True 

176 

177 if is_variable and is_number: 

178 raise ValueError( 

179 'if one argument is Tensor, all arguments should be Tensor' 

180 ) 

181 

182 return is_variable

184 def _to_tensor(self, *args): 

185 """ 

186 Argument convert args to Tensor 

187 

188 Args: 

189 value (float, list, numpy.ndarray, Tensor) 

190 Returns: 

191 Tensor of args. 

192 """ 

193 numpy_args = [] 

194 variable_args = [] 

195 tmp = 0.0 

196 

197 for arg in args: 

198 if isinstance(arg, float): 

199 arg = [arg] 

200 if not isinstance(arg, (list, tuple, np.ndarray, tensor.Variable)): 

201 raise TypeError( 

202 "Type of input args must be float, list, numpy.ndarray or Tensor, but received type {}".format( 

203 type(arg) 

204 ) 

205 ) 

206 

207 arg_np = np.array(arg) 

208 arg_dtype = arg_np.dtype 

209 if str(arg_dtype) != 'float32': 

210 if str(arg_dtype) != 'float64': 

211 # "assign" op doesn't support float64. if dtype is float64, float32 variable will be generated 

212 # and converted to float64 later using "cast". 

213 warnings.warn( 

214 "data type of argument only support float32 and float64, your argument will be convert to float32." 

215 ) 

216 arg_np = arg_np.astype('float32') 

217 # tmp is used to support broadcast, it summarizes shapes of all the args and get the mixed shape. 

218 tmp = tmp + arg_np 

219 numpy_args.append(arg_np) 

220 

221 dtype = tmp.dtype 

222 for arg in numpy_args: 

223 arg_broadcasted, _ = np.broadcast_arrays(arg, tmp) 

224 arg_variable = tensor.create_tensor(dtype=dtype) 

225 tensor.assign(arg_broadcasted, arg_variable) 

226 variable_args.append(arg_variable) 

227 

228 return tuple(variable_args)

230 def _check_values_dtype_in_probs(self, param, value): 

231 """ 

232 Log_prob and probs methods have input ``value``, if value's dtype is different from param, 

233 convert value's dtype to be consistent with param's dtype. 

234 

235 Args: 

236 param (Tensor): low and high in Uniform class, loc and scale in Normal class. 

237 value (Tensor): The input tensor. 

238 

239 Returns: 

240 value (Tensor): Change value's dtype if value's dtype is different from param. 

241 """ 

242 if _non_static_mode(): 

243 if value.dtype != param.dtype and convert_dtype(value.dtype) in [ 

244 'float32', 

245 'float64', 

246 ]: 

247 warnings.warn( 

248 "dtype of input 'value' needs to be the same as parameters of distribution class. dtype of 'value' will be converted." 

249 ) 

250 if in_dygraph_mode(): 

251 return _C_ops.cast(value, param.dtype) 

252 if _in_legacy_dygraph(): 

253 return _legacy_C_ops.cast( 

254 value, 'in_dtype', value.dtype, 'out_dtype', param.dtype 

255 ) 

256 return value 

257 

258 check_variable_and_dtype( 

259 value, 'value', ['float32', 'float64'], 'log_prob' 

260 ) 

261 if value.dtype != param.dtype: 

262 warnings.warn( 

263 "dtype of input 'value' needs to be the same as parameters of distribution class. dtype of 'value' will be converted." 

264 ) 

265 return tensor.cast(value, dtype=param.dtype) 

266 return value

268 def _probs_to_logits(self, probs, is_binary=False): 

269 r""" 

270 Converts probabilities into logits. For the binary, probs denotes the 

271 probability of occurrence of the event indexed by `1`. For the 

272 multi-dimensional, values of last axis denote the probabilities of 

273 occurrence of each of the events. 

274 """ 

275 return ( 

276 (paddle.log(probs) - paddle.log1p(-probs)) 

277 if is_binary 

278 else paddle.log(probs) 

279 )

281 def _logits_to_probs(self, logits, is_binary=False): 

282 r""" 

283 Converts logits into probabilities. For the binary, each value denotes 

284 log odds, whereas for the multi-dimensional case, the values along the 

285 last dimension denote the log probabilities of the events. 

286 """ 

287 return ( 

288 paddle.nn.functional.sigmoid(logits) 

289 if is_binary 

290 else paddle.nn.functional.softmax(logits, axis=-1) 

291 )
