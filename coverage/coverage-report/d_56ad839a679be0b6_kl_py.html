32def kl_divergence(p, q): 

33 r""" 

34 Kullback-Leibler divergence between distribution p and q. 

35 

36 .. math:: 

37 

38 KL(p||q) = \int p(x)log\frac{p(x)}{q(x)} \mathrm{d}x 

39 

40 Args: 

41 p (Distribution): ``Distribution`` object. Inherits from the Distribution Base class. 

42 q (Distribution): ``Distribution`` object. Inherits from the Distribution Base class. 

43 

44 Returns: 

45 Tensor, Batchwise KL-divergence between distribution p and q. 

46 

47 Examples: 

48 

49 .. code-block:: python 

50 

51 import paddle 

52 

53 p = paddle.distribution.Beta(alpha=0.5, beta=0.5) 

54 q = paddle.distribution.Beta(alpha=0.3, beta=0.7) 

55 

56 print(paddle.distribution.kl_divergence(p, q)) 

57 # Tensor(shape=[1], dtype=float32, place=CUDAPlace(0), stop_gradient=True, 

58 # [0.21193528]) 

59 

60 """ 

61 return _dispatch(type(p), type(q))(p, q)

64def register_kl(cls_p, cls_q): 

65 """Decorator for register a KL divergence implemention function. 

66 

67 The ``kl_divergence(p, q)`` function will search concrete implemention 

68 functions registered by ``register_kl``, according to multi-dispatch pattern. 

69 If an implemention function is found, it will return the result, otherwise, 

70 it will raise ``NotImplementError`` exception. Users can register 

71 implemention funciton by the decorator. 

72 

73 Args: 

74 cls_p (Distribution): The Distribution type of Instance p. Subclass derived from ``Distribution``. 

75 cls_q (Distribution): The Distribution type of Instance q. Subclass derived from ``Distribution``. 

76 

77 Examples: 

78 .. code-block:: python 

79 

80 import paddle 

81 

82 @paddle.distribution.register_kl(paddle.distribution.Beta, paddle.distribution.Beta) 

83 def kl_beta_beta(): 

84 pass # insert implementation here 

85 """ 

86 if not issubclass(cls_p, Distribution) or not issubclass( 86 ↛ 89line 86 didn't jump to line 89, because the condition on line 86 was never true

87 cls_q, Distribution 

88 ): 

89 raise TypeError('cls_p and cls_q must be subclass of Distribution') 

90 

91 def decorator(f): 

92 _REGISTER_TABLE[cls_p, cls_q] = f 

93 return f 

94 

95 return decorator

98def _dispatch(cls_p, cls_q): 

99 """Multiple dispatch into concrete implement function""" 

100 

101 # find all matched super class pair of p and q 

102 matchs = [ 

103 (super_p, super_q) 

104 for super_p, super_q in _REGISTER_TABLE 

105 if issubclass(cls_p, super_p) and issubclass(cls_q, super_q) 

106 ] 

107 if not matchs: 

108 raise NotImplementedError 

109 

110 left_p, left_q = min(_Compare(*m) for m in matchs).classes 

111 right_p, right_q = min(_Compare(*reversed(m)) for m in matchs).classes 

112 

113 if _REGISTER_TABLE[left_p, left_q] is not _REGISTER_TABLE[right_p, right_q]: 

114 warnings.warn( 

115 'Ambiguous kl_divergence({}, {}). Please register_kl({}, {})'.format( 

116 cls_p.__name__, 

117 cls_q.__name__, 

118 left_p.__name__, 

119 right_q.__name__, 

120 ), 

121 RuntimeWarning, 

122 ) 

123 

124 return _REGISTER_TABLE[left_p, left_q]

127@functools.total_ordering 

128class _Compare(object): 

129 def __init__(self, *classes): 

130 self.classes = classes 

131 

132 def __eq__(self, other): 

133 return self.classes == other.classes 

134 

135 def __le__(self, other): 

136 for cls_x, cls_y in zip(self.classes, other.classes): 

137 if not issubclass(cls_x, cls_y): 

138 return False 

139 if cls_x is not cls_y: 

140 break 

141 return True

144@register_kl(Beta, Beta) 

145def _kl_beta_beta(p, q): 

146 return ( 

147 (q.alpha.lgamma() + q.beta.lgamma() + (p.alpha + p.beta).lgamma()) 

148 - (p.alpha.lgamma() + p.beta.lgamma() + (q.alpha + q.beta).lgamma()) 

149 + ((p.alpha - q.alpha) * p.alpha.digamma()) 

150 + ((p.beta - q.beta) * p.beta.digamma()) 

151 + ( 

152 ((q.alpha + q.beta) - (p.alpha + p.beta)) 

153 * (p.alpha + p.beta).digamma() 

154 ) 

155 )

158@register_kl(Dirichlet, Dirichlet) 

159def _kl_dirichlet_dirichlet(p, q): 

160 return ( 

161 (p.concentration.sum(-1).lgamma() - q.concentration.sum(-1).lgamma()) 

162 - ((p.concentration.lgamma() - q.concentration.lgamma()).sum(-1)) 

163 + ( 

164 ( 

165 (p.concentration - q.concentration) 

166 * ( 

167 p.concentration.digamma() 

168 - p.concentration.sum(-1).digamma().unsqueeze(-1) 

169 ) 

170 ).sum(-1) 

171 ) 

172 )

190@register_kl(ExponentialFamily, ExponentialFamily) 

191def _kl_expfamily_expfamily(p, q): 

192 """Compute kl-divergence using `Bregman divergences <https://www.lix.polytechnique.fr/~nielsen/EntropyEF-ICIP2010.pdf>`_""" 

193 if not type(p) == type(q): 

194 raise NotImplementedError 

195 

196 p_natural_params = [] 

197 for param in p._natural_parameters: 

198 param = param.detach() 

199 param.stop_gradient = False 

200 p_natural_params.append(param) 

201 

202 q_natural_params = q._natural_parameters 

203 

204 p_log_norm = p._log_normalizer(*p_natural_params) 

205 

206 try: 

207 if _non_static_mode(): 

208 p_grads = paddle.grad( 

209 p_log_norm, p_natural_params, create_graph=True 

210 ) 

211 else: 

212 p_grads = paddle.static.gradients(p_log_norm, p_natural_params) 

213 except RuntimeError as e: 

214 raise TypeError( 

215 "Cann't compute kl_divergence({cls_p}, {cls_q}) use bregman divergence. Please register_kl({cls_p}, {cls_q}).".format( 

216 cls_p=type(p).__name__, cls_q=type(q).__name__ 

217 ) 

218 ) from e 

219 

220 kl = q._log_normalizer(*q_natural_params) - p_log_norm 

221 for p_param, q_param, p_grad in zip( 

222 p_natural_params, q_natural_params, p_grads 

223 ): 

224 term = (q_param - p_param) * p_grad 

225 kl -= _sum_rightmost(term, len(q.event_shape)) 

226 

227 return kl

230def _sum_rightmost(value, n): 

231 return value.sum(list(range(-n, 0))) if n > 0 else value
