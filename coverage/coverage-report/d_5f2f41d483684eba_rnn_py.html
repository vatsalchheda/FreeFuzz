49def split_states(states, bidirectional=False, state_components=1): 

50 r""" 

51 Split states of RNN network into possibly nested list or tuple of 

52 states of each RNN cells of the RNN network. 

53 

54 Parameters: 

55 states (Tensor|tuple|list): the concatenated states for RNN network. 

56 When `state_components` is 1, states in a Tensor with shape 

57 `(L*D, N, C)` where `L` is the number of layers of the RNN  

58 network, `D` is the number of directions of the RNN network(1  

59 for unidirectional RNNs and 2 for bidirectional RNNs), `N` is  

60 the batch size of the input to the RNN network, `C` is the  

61 hidden size of the RNN network.  

62 

63 When `state_components` is larger than 1, `states` is a tuple of  

64 `state_components` Tensors that meet the requirements described  

65 above.  

66  

67 For SimpleRNNs and GRUs, `state_components` is 1, and for LSTMs,  

68 `state_components` is 2. 

69 bidirectional (bool): whether the state is of a bidirectional RNN  

70 network. Defaults to False. 

71 state_components (int): the number of the components of the states. see 

72 `states` above. Defaults to 1. 

73  

74 Returns: 

75 A nested list or tuple of RNN cell states.  

76 If `bidirectional` is True, it can be indexed twice to get an RNN  

77 cell state. The first index indicates the layer, the second index  

78 indicates the direction. 

79 If `bidirectional` is False, it can be indexed once to get an RNN 

80 cell state. The index indicates the layer. 

81 Note that if `state_components` is larger than 1, an RNN cell state 

82 can be indexed one more time to get a tensor of shape(N, C), where  

83 `N` is the batch size of the input to the RNN cell, and `C` is the 

84 hidden size of the RNN cell. 

85 """ 

86 if state_components == 1: 

87 states = paddle.unstack(states) 

88 if not bidirectional: 

89 return states 

90 else: 

91 return list(zip(states[::2], states[1::2])) 

92 else: 

93 assert len(states) == state_components 

94 states = tuple([paddle.unstack(item) for item in states]) 

95 if not bidirectional: 

96 return list(zip(*states)) 

97 else: 

98 states = list(zip(*states)) 

99 return list(zip(states[::2], states[1::2]))

102def concat_states(states, bidirectional=False, state_components=1): 

103 r""" 

104 Concatenate a possibly nested list or tuple of RNN cell states into a  

105 compact form. 

106 

107 Parameters: 

108 states (list|tuple): a possibly nested list or tuple of RNN cell  

109 states.  

110 If `bidirectional` is True, it can be indexed twice to get an  

111 RNN cell state. The first index indicates the layer, the second  

112 index indicates the direction. 

113 If `bidirectional` is False, it can be indexed once to get an RNN 

114 cell state. The index indicates the layer. 

115 Note that if `state_components` is larger than 1, an RNN cell  

116 state can be indexed one more time to get a tensor of shape(N, C),  

117 where `N` is the batch size of the input to the RNN cell, and  

118 `C` is the hidden size of the RNN cell.  

119 bidirectional (bool): whether the state is of a bidirectional RNN  

120 network. Defaults to False. 

121 state_components (int): the number of the components of the states. see 

122 `states` above. Defaults to 1. 

123  

124 Returns: 

125 Concatenated states for RNN network. 

126 When `state_components` is 1, states in a Tensor with shape 

127 `(L\*D, N, C)` where `L` is the number of layers of the RNN  

128 network, `D` is the number of directions of the RNN network(1 for  

129 unidirectional RNNs and 2 for bidirectional RNNs), `N` is the batch  

130 size of the input to the RNN network, `C` is the hidden size of the  

131 RNN network. 

132  

133 """ 

134 if state_components == 1: 

135 return paddle.stack(flatten(states)) 

136 else: 

137 states = flatten(states) 

138 componnets = [] 

139 for i in range(state_components): 

140 componnets.append(states[i::state_components]) 

141 return tuple([paddle.stack(item) for item in componnets])

271class SimpleRNNCell(RNNCellBase): 

272 r""" 

273 Elman RNN (SimpleRNN) cell. Given the inputs and previous states, it  

274 computes the outputs and updates states. 

275 

276 The formula used is as follows: 

277 

278 .. math:: 

279 h_{t} & = act(W_{ih}x_{t} + b_{ih} + W_{hh}h_{t-1} + b_{hh}) 

280 

281 y_{t} & = h_{t} 

282  

283 where :math:`act` is for :attr:`activation`. 

284 

285 Please refer to `Finding Structure in Time  

286 <https://crl.ucsd.edu/~elman/Papers/fsit.pdf>`_ for more details. 

287  

288 Parameters: 

289 input_size (int): The input size. 

290 hidden_size (int): The hidden size. 

291 activation (str, optional): The activation in the SimpleRNN cell.  

292 It can be `tanh` or `relu`. Defaults to `tanh`. 

293 weight_ih_attr (ParamAttr, optional): The parameter attribute for  

294 :math:`weight_ih`. Default: None. 

295 weight_hh_attr(ParamAttr, optional): The parameter attribute for  

296 :math:`weight_hh`. Default: None. 

297 bias_ih_attr (ParamAttr, optional): The parameter attribute for the  

298 :math:`bias_ih`. Default: None. 

299 bias_hh_attr (ParamAttr, optional): The parameter attribute for the  

300 :math:`bias_hh`. Default: None. 

301 name (str, optional): Name for the operation (optional, default is  

302 None). For more information, please refer to :ref:`api_guide_Name`. 

303 

304 Variables: 

305 - **weight_ih** (Parameter): shape (hidden_size, input_size), input to hidden weight, corresponding to :math:`W_{ih}` in the formula. 

306 - **weight_hh** (Parameter): shape (hidden_size, hidden_size), hidden to hidden weight, corresponding to :math:`W_{hh}` in the formula. 

307 - **bias_ih** (Parameter): shape (hidden_size, ), input to hidden bias, corresponding to :math:`b_{ih}` in the formula. 

308 - **bias_hh** (Parameter): shape (hidden_size, ), hidden to hidden bias, corresponding to :math:`b_{hh}` in the formula. 

309  

310 Inputs: 

311 - **inputs** (Tensor): shape `[batch_size, input_size]`, the input, corresponding to :math:`x_{t}` in the formula. 

312 - **states** (Tensor, optional): shape `[batch_size, hidden_size]`, the previous hidden state, corresponding to :math:`h_{t-1}` in the formula. When states is None, zero state is used. Defaults to None. 

313 

314 Returns: 

315 - **outputs** (Tensor): shape `[batch_size, hidden_size]`, the output, corresponding to :math:`h_{t}` in the formula. 

316 - **states** (Tensor): shape `[batch_size, hidden_size]`, the new hidden state, corresponding to :math:`h_{t}` in the formula. 

317  

318 Notes: 

319 All the weights and bias are initialized with `Uniform(-std, std)` by default. Where std = :math:`\frac{1}{\sqrt{hidden\_size}}`. For more information about parameter initialization, please refer to :ref:`api_fluid_ParamAttr`. 

320 

321 Examples: 

322 

323 .. code-block:: python 

324 

325 import paddle 

326 

327 x = paddle.randn((4, 16)) 

328 prev_h = paddle.randn((4, 32)) 

329 

330 cell = paddle.nn.SimpleRNNCell(16, 32) 

331 y, h = cell(x, prev_h) 

332 print(y.shape) 

333 

334 #[4,32] 

335 

336 """ 

337 

338 def __init__(self, 

339 input_size, 

340 hidden_size, 

341 activation="tanh", 

342 weight_ih_attr=None, 

343 weight_hh_attr=None, 

344 bias_ih_attr=None, 

345 bias_hh_attr=None, 

346 name=None): 

347 super(SimpleRNNCell, self).__init__() 

348 if hidden_size <= 0: 

349 raise ValueError( 

350 "hidden_size of {} must be greater than 0, but now equals to {}" 

351 .format(self.__class__.__name__, hidden_size)) 

352 std = 1.0 / math.sqrt(hidden_size) 

353 self.weight_ih = self.create_parameter( 

354 (hidden_size, input_size), 

355 weight_ih_attr, 

356 default_initializer=I.Uniform(-std, std)) 

357 self.weight_hh = self.create_parameter( 

358 (hidden_size, hidden_size), 

359 weight_hh_attr, 

360 default_initializer=I.Uniform(-std, std)) 

361 self.bias_ih = self.create_parameter( 

362 (hidden_size, ), 

363 bias_ih_attr, 

364 is_bias=True, 

365 default_initializer=I.Uniform(-std, std)) 

366 self.bias_hh = self.create_parameter( 

367 (hidden_size, ), 

368 bias_hh_attr, 

369 is_bias=True, 

370 default_initializer=I.Uniform(-std, std)) 

371 

372 self.input_size = input_size 

373 self.hidden_size = hidden_size 

374 if activation not in ["tanh", "relu"]: 

375 raise ValueError( 

376 "activation for SimpleRNNCell should be tanh or relu, " 

377 "but get {}".format(activation))

379 self._activation_fn = paddle.tanh \ 

380 if activation == "tanh" \

385 states = self.get_initial_states(inputs, self.state_shape)

387 i2h = paddle.matmul(inputs, self.weight_ih, transpose_y=True)

391 if self.bias_hh is not None: 

392 h2h += self.bias_hh 

393 h = self._activation_fn(i2h + h2h) 

394 return h, h

407class LSTMCell(RNNCellBase): 

408 r""" 

409 Long-Short Term Memory(LSTM) RNN cell. Given the inputs and previous states,  

410 it computes the outputs and updates states. 

411 

412 The formula used is as follows: 

413 

414 .. math:: 

415 i_{t} & = \sigma(W_{ii}x_{t} + b_{ii} + W_{hi}h_{t-1} + b_{hi}) 

416 

417 f_{t} & = \sigma(W_{if}x_{t} + b_{if} + W_{hf}h_{t-1} + b_{hf}) 

418 

419 o_{t} & = \sigma(W_{io}x_{t} + b_{io} + W_{ho}h_{t-1} + b_{ho}) 

420 

421 \widetilde{c}_{t} & = \tanh (W_{ig}x_{t} + b_{ig} + W_{hg}h_{t-1} + b_{hg}) 

422 

423 c_{t} & = f_{t} * c_{t-1} + i_{t} * \widetilde{c}_{t} 

424 

425 h_{t} & = o_{t} * \tanh(c_{t}) 

426 

427 y_{t} & = h_{t} 

428 

429 where :math:`\sigma` is the sigmoid fucntion, and * is the elemetwise  

430 multiplication operator. 

431 

432 Please refer to `An Empirical Exploration of Recurrent Network Architectures 

433 <http://proceedings.mlr.press/v37/jozefowicz15.pdf>`_ for more details. 

434 

435 Parameters: 

436 input_size (int): The input size. 

437 hidden_size (int): The hidden size. 

438 weight_ih_attr(ParamAttr, optional): The parameter attribute for  

439 `weight_ih`. Default: None. 

440 weight_hh_attr(ParamAttr, optional): The parameter attribute for  

441 `weight_hh`. Default: None. 

442 bias_ih_attr (ParamAttr, optional): The parameter attribute for the  

443 `bias_ih`. Default: None. 

444 bias_hh_attr (ParamAttr, optional): The parameter attribute for the  

445 `bias_hh`. Default: None. 

446 name (str, optional): Name for the operation (optional, default is  

447 None). For more information, please refer to :ref:`api_guide_Name`. 

448 

449 Variables: 

450 - **weight_ih** (Parameter): shape (4 * hidden_size, input_size), input to hidden weight, which corresponds to the concatenation of :math:`W_{ii}, W_{if}, W_{ig}, W_{io}` in the formula. 

451 - **weight_hh** (Parameter): shape (4 * hidden_size, hidden_size), hidden to hidden weight, which corresponds to the concatenation of :math:`W_{hi}, W_{hf}, W_{hg}, W_{ho}` in the formula. 

452 - **bias_ih** (Parameter): shape (4 * hidden_size, ), input to hidden bias, which corresponds to the concatenation of :math:`b_{ii}, b_{if}, b_{ig}, b_{io}` in the formula. 

453 - **bias_hh** (Parameter): shape (4 * hidden_size, ), hidden to hidden bias, swhich corresponds to the concatenation of :math:`b_{hi}, b_{hf}, b_{hg}, b_{ho}` in the formula. 

454 

455 Inputs: 

456 - **inputs** (Tensor): shape `[batch_size, input_size]`, the input, corresponding to :math:`x_t` in the formula. 

457 - **states** (list|tuple, optional): a list/tuple of two tensors, each of shape `[batch_size, hidden_size]`, the previous hidden state, corresponding to :math:`h_{t-1}, c_{t-1}` in the formula. When states is None, zero state is used. Defaults to None. 

458 

459 Returns: 

460 - **outputs** (Tensor): shape `[batch_size, hidden_size]`, the output, corresponding to :math:`h_{t}` in the formula. 

461 - **states** (tuple): a tuple of two tensors, each of shape `[batch_size, hidden_size]`, the new hidden states, corresponding to :math:`h_{t}, c_{t}` in the formula. 

462 

463 Notes: 

464 All the weights and bias are initialized with `Uniform(-std, std)` by  

465 default. Where std = :math:`\frac{1}{\sqrt{hidden\_size}}`. For more  

466 information about parameter initialization, please refer to :ref:`api_fluid_ParamAttr`. 

467 

468 Examples: 

469 

470 .. code-block:: python 

471 

472 import paddle 

473 

474 x = paddle.randn((4, 16)) 

475 prev_h = paddle.randn((4, 32)) 

476 prev_c = paddle.randn((4, 32)) 

477 

478 cell = paddle.nn.LSTMCell(16, 32) 

479 y, (h, c) = cell(x, (prev_h, prev_c)) 

480 

481 print(y.shape) 

482 print(h.shape) 

483 print(c.shape) 

484 

485 #[4,32] 

486 #[4,32] 

487 #[4,32] 

488 

489 """ 

490 

491 def __init__(self, 

492 input_size, 

493 hidden_size, 

494 weight_ih_attr=None, 

495 weight_hh_attr=None, 

496 bias_ih_attr=None, 

497 bias_hh_attr=None, 

498 name=None): 

499 super(LSTMCell, self).__init__() 

500 if hidden_size <= 0: 

501 raise ValueError( 

502 "hidden_size of {} must be greater than 0, but now equals to {}" 

503 .format(self.__class__.__name__, hidden_size)) 

504 std = 1.0 / math.sqrt(hidden_size) 

505 self.weight_ih = self.create_parameter( 

506 (4 * hidden_size, input_size), 

507 weight_ih_attr, 

508 default_initializer=I.Uniform(-std, std)) 

509 self.weight_hh = self.create_parameter( 

510 (4 * hidden_size, hidden_size), 

511 weight_hh_attr, 

512 default_initializer=I.Uniform(-std, std)) 

513 self.bias_ih = self.create_parameter( 

514 (4 * hidden_size, ), 

515 bias_ih_attr, 

516 is_bias=True, 

517 default_initializer=I.Uniform(-std, std)) 

518 self.bias_hh = self.create_parameter( 

519 (4 * hidden_size, ), 

520 bias_hh_attr, 

521 is_bias=True, 

522 default_initializer=I.Uniform(-std, std)) 

523 

524 self.hidden_size = hidden_size 

525 self.input_size = input_size 

526 self._gate_activation = F.sigmoid 

527 self._activation = paddle.tanh 

528 

529 def forward(self, inputs, states=None): 

530 if states is None: 

531 states = self.get_initial_states(inputs, self.state_shape) 

532 pre_hidden, pre_cell = states 

533 gates = paddle.matmul(inputs, self.weight_ih, transpose_y=True) 

534 if self.bias_ih is not None: 

535 gates = gates + self.bias_ih 

536 gates += paddle.matmul(pre_hidden, self.weight_hh, transpose_y=True) 

537 if self.bias_hh is not None: 

538 gates = gates + self.bias_hh 

539 

540 chunked_gates = paddle.split(gates, num_or_sections=4, axis=-1) 

541 

542 i = self._gate_activation(chunked_gates[0]) 

543 f = self._gate_activation(chunked_gates[1]) 

544 o = self._gate_activation(chunked_gates[3]) 

545 c = f * pre_cell + i * self._activation(chunked_gates[2]) 

546 h = o * self._activation(c) 

547 

548 return h, (h, c)

564class GRUCell(RNNCellBase): 

565 r""" 

566 Gated Recurrent Unit (GRU) RNN cell. Given the inputs and previous states,  

567 it computes the outputs and updates states. 

568 

569 The formula for GRU used is as follows: 

570 

571 .. math:: 

572 

573 r_{t} & = \sigma(W_{ir}x_{t} + b_{ir} + W_{hr}h_{t-1} + b_{hr}) 

574 

575 z_{t} & = \sigma(W_{iz}x_{t} + b_{iz} + W_{hz}h_{t-1} + b_{hz}) 

576 

577 \widetilde{h}_{t} & = \tanh(W_{ic}x_{t} + b_{ic} + r_{t} * (W_{hc}h_{t-1} + b_{hc})) 

578 

579 h_{t} & = z_{t} * h_{t-1} + (1 - z_{t}) * \widetilde{h}_{t} 

580 

581 y_{t} & = h_{t} 

582  

583 where :math:`\sigma` is the sigmoid fucntion, and * is the elemetwise  

584 multiplication operator. 

585 

586 Please refer to `An Empirical Exploration of Recurrent Network Architectures 

587 <http://proceedings.mlr.press/v37/jozefowicz15.pdf>`_ for more details. 

588 

589 Parameters: 

590 input_size (int): The input size. 

591 hidden_size (int): The hidden size. 

592 weight_ih_attr(ParamAttr, optional): The parameter attribute for  

593 `weight_ih`. Default: None. 

594 weight_hh_attr(ParamAttr, optional): The parameter attribute for  

595 `weight_hh`. Default: None. 

596 bias_ih_attr (ParamAttr, optional): The parameter attribute for the  

597 `bias_ih`. Default: None. 

598 bias_hh_attr (ParamAttr, optional): The parameter attribute for the  

599 `bias_hh`. Default: None. 

600 name (str, optional): Name for the operation (optional, default is  

601 None). For more information, please refer to :ref:`api_guide_Name`. 

602 

603 Variables: 

604 - **weight_ih** (Parameter): shape (3 * hidden_size, input_size), input to hidden weight, which corresponds to the concatenation of :math:`W_{ir}, W_{iz}, W_{ic}` in the formula. 

605 - **weight_hh** (Parameter): shape (3 * hidden_size, hidden_size), hidden to hidden weight, which corresponds to the concatenation of :math:`W_{hr}, W_{hz}, W_{hc}` in the formula. 

606 - **bias_ih** (Parameter): shape (3 * hidden_size, ), input to hidden bias, which corresponds to the concatenation of :math:`b_{ir}, b_{iz}, b_{ic}` in the formula. 

607 - **bias_hh** (Parameter): shape (3 * hidden_size, ), hidden to hidden bias, swhich corresponds to the concatenation of :math:`b_{hr}, b_{hz}, b_{hc}` in the formula. 

608 

609 Inputs: 

610 - **inputs** (Tensor): A tensor with shape `[batch_size, input_size]`, corresponding to :math:`x_t` in the formula. 

611 - **states** (Tensor): A tensor with shape `[batch_size, hidden_size]`, corresponding to :math:`h_{t-1}` in the formula. 

612 

613 Returns: 

614 - **outputs** (Tensor): shape `[batch_size, hidden_size]`, the output, corresponding to :math:`h_{t}` in the formula. 

615 - **states** (Tensor): shape `[batch_size, hidden_size]`, the new hidden state, corresponding to :math:`h_{t}` in the formula. 

616  

617 Notes: 

618 All the weights and bias are initialized with `Uniform(-std, std)` by  

619 default. Where std = :math:`\frac{1}{\sqrt{hidden\_size}}`. For more  

620 information about parameter initialization, please refer to s:ref:`api_fluid_ParamAttr`. 

621 

622 Examples:

635 print(h.shape) 

636 

637 #[4,32] 

638 #[4,32] 

639 

640 """

642 def __init__(self, 

643 input_size, 

644 hidden_size, 

645 weight_ih_attr=None, 

646 weight_hh_attr=None, 

647 bias_ih_attr=None, 

648 bias_hh_attr=None, 

649 name=None): 

650 super(GRUCell, self).__init__() 

651 if hidden_size <= 0: 

652 raise ValueError( 

653 "hidden_size of {} must be greater than 0, but now equals to {}" 

654 .format(self.__class__.__name__, hidden_size)) 

655 std = 1.0 / math.sqrt(hidden_size) 

656 self.weight_ih = self.create_parameter( 

657 (3 * hidden_size, input_size), 

658 weight_ih_attr, 

659 default_initializer=I.Uniform(-std, std)) 

660 self.weight_hh = self.create_parameter( 

661 (3 * hidden_size, hidden_size), 

662 weight_hh_attr, 

663 default_initializer=I.Uniform(-std, std)) 

664 self.bias_ih = self.create_parameter( 

665 (3 * hidden_size, ), 

666 bias_ih_attr, 

667 is_bias=True, 

668 default_initializer=I.Uniform(-std, std)) 

669 self.bias_hh = self.create_parameter( 

670 (3 * hidden_size, ), 

671 bias_hh_attr, 

672 is_bias=True, 

673 default_initializer=I.Uniform(-std, std)) 

674 

675 self.hidden_size = hidden_size 

676 self.input_size = input_size 

677 self._gate_activation = F.sigmoid 

678 self._activation = paddle.tanh 

679 

680 def forward(self, inputs, states=None): 

681 if states is None: 

682 states = self.get_initial_states(inputs, self.state_shape) 

683 

684 pre_hidden = states 

685 x_gates = paddle.matmul(inputs, self.weight_ih, transpose_y=True) 

686 if self.bias_ih is not None: 

687 x_gates = x_gates + self.bias_ih 

688 h_gates = paddle.matmul(pre_hidden, self.weight_hh, transpose_y=True) 

689 if self.bias_hh is not None: 

690 h_gates = h_gates + self.bias_hh 

691 

692 x_r, x_z, x_c = paddle.split(x_gates, num_or_sections=3, axis=1) 

693 h_r, h_z, h_c = paddle.split(h_gates, num_or_sections=3, axis=1) 

694 

695 r = self._gate_activation(x_r + h_r) 

696 z = self._gate_activation(x_z + h_z) 

697 c = self._activation(x_c + r * h_c) # apply reset gate after mm 

698 h = (pre_hidden - c) * z + c 

699 

700 return h, h
