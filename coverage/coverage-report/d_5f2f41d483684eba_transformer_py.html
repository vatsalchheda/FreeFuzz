109class MultiHeadAttention(Layer): 

110 """ 

111 Attention mapps queries and a set of key-value pairs to outputs, and 

112 Multi-Head Attention performs multiple parallel attention to jointly attending 

113 to information from different representation subspaces. 

114 

115 Please refer to `Attention Is All You Need <https://arxiv.org/pdf/1706.03762.pdf>`_ 

116 for more details. 

117 

118 Parameters: 

119 embed_dim (int): The expected feature size in the input and output. 

120 num_heads (int): The number of heads in multi-head attention. 

121 dropout (float, optional): The dropout probability used on attention 

122 weights to drop some attention targets. 0 for no dropout. Default 0 

123 kdim (int, optional): The feature size in key. If None, assumed equal to 

124 `embed_dim`. Default None. 

125 vdim (int, optional): The feature size in value. If None, assumed equal to 

126 `embed_dim`. Default None. 

127 need_weights (bool, optional): Indicate whether to return the attention 

128 weights. Default False. 

129 weight_attr(ParamAttr, optional): To specify the weight parameter property. 

130 Default: None, which means the default weight parameter property is used. 

131 See usage for details in :code:`ParamAttr` . 

132 bias_attr (ParamAttr|bool, optional): To specify the bias parameter property. 

133 Default: None, which means the default bias parameter property is used. 

134 If it is set to False, this layer will not have trainable bias parameter. 

135 See usage for details in :code:`ParamAttr` . 

136  

137 Examples: 

138 

139 .. code-block:: python 

140 

141 import paddle 

142 

143 # encoder input: [batch_size, sequence_length, d_model] 

144 query = paddle.rand((2, 4, 128)) 

145 # self attention mask: [batch_size, num_heads, query_len, query_len] 

146 attn_mask = paddle.rand((2, 2, 4, 4)) 

147 multi_head_attn = paddle.nn.MultiHeadAttention(128, 2) 

148 output = multi_head_attn(query, None, None, attn_mask=attn_mask) # [2, 4, 128] 

149 """

197 def _prepare_qkv(self, query, key, value, cache=None): 

198 r""" 

199 Prapares linear projected queries, keys and values for usage of subsequnt 

200 multiple parallel attention. If `cache` is not None, using cached results 

201 to reduce redundant calculations. 

202 

203 Parameters: 

204 query (Tensor): The queries for multi-head attention. It is a 

205 tensor with shape `[batch_size, query_length, embed_dim]`. The 

206 data type should be float32 or float64. 

207 key (Tensor): The keys for multi-head attention. It is 

208 a tensor with shape `[batch_size, key_length, kdim]`. The 

209 data type should be float32 or float64. If None, use `query` as 

210 `key`. 

211 value (Tensor): The values for multi-head attention. It 

212 is a tensor with shape `[batch_size, value_length, vdim]`. 

213 The data type should be float32 or float64. If None, use `query` as 

214 `value`. 

215 cache (MultiHeadAttention.Cache|MultiHeadAttention.StaticCache, optional): 

216 It is a namedtuple with `k` and `v` as fields, and stores tensors 

217 shaped `[batch_size, num_heads, length, embed_dim]` which are results 

218 of linear projection, reshape and transpose calculations in 

219 MultiHeadAttention. If is an instance of `Cache`, `k` and `v` 

220 fields reserve intermediate results of previous positions, which 

221 mostly used for decoder self attention. If it is an instance of 

222 `StaticCache`, `key` and `value` args would be ignored, `k` and 

223 `v` fields would be used as calculated results on `key` and 

224 `value`, which mostly used for decoder-encoder cross attention. 

225 It is only used for inference and should be None for training. 

226 Default None. 

227 

228 Returns: 

229 tuple: A tuple including linear projected keys and values. These two \ 

230 tensors have shapes `[batch_size, n_head, sequence_length, d_key]` \ 

231 and `[batch_size, n_head, sequence_length, d_value]` separately, \ 

232 and their data types are same as inputs. 

233 """ 

234 q = self.q_proj(query) 

235 q = tensor.reshape(x=q, shape=[0, 0, self.num_heads, self.head_dim]) 

236 q = tensor.transpose(x=q, perm=[0, 2, 1, 3]) 

237 

238 if isinstance(cache, self.StaticCache): 

239 # for encoder-decoder attention in inference and has cached 

240 k, v = cache.k, cache.v 

241 else: 

242 k, v = self.compute_kv(key, value) 

243 

244 if isinstance(cache, self.Cache): 

245 # for decoder self-attention in inference 

246 k = tensor.concat([cache.k, k], axis=2) 

247 v = tensor.concat([cache.v, v], axis=2) 

248 cache = self.Cache(k, v) 

249 

250 return (q, k, v) if cache is None else (q, k, v, cache)

252 def compute_kv(self, key, value): 

253 r""" 

254 Applies linear projection on input keys and values, then splits heads 

255 (reshape and transpose) to get keys and values from different representation 

256 subspaces. The results are used as key-values pairs for subsequent multiple 

257 parallel attention. 

258  

259 It is part of calculations in multi-head attention, and is provided as 

260 a method to pre-compute and prefetch these results, thus we can use them 

261 to construct cache for inference. 

262 

263 Parameters: 

264 key (Tensor): The keys for multi-head attention. It is a tensor 

265 with shape `[batch_size, sequence_length, kdim]`. The data type 

266 should be float32 or float64. 

267 value (Tensor): The values for multi-head attention. It is a tensor 

268 with shape `[batch_size, sequence_length, vdim]`. The data type 

269 should be float32 or float64. 

270 

271 Returns: 

272 tuple: A tuple including transformed keys and values. Their shapes \ 

273 both are `[batch_size, num_heads, sequence_length, embed_dim // num_heads]`, \ 

274 and their data types are same as inputs. 

275 """ 

276 k = self.k_proj(key) 

277 v = self.v_proj(value) 

278 k = tensor.reshape(x=k, shape=[0, 0, self.num_heads, self.head_dim]) 

279 k = tensor.transpose(x=k, perm=[0, 2, 1, 3]) 

280 v = tensor.reshape(x=v, shape=[0, 0, self.num_heads, self.head_dim]) 

281 v = tensor.transpose(x=v, perm=[0, 2, 1, 3]) 

282 return k, v

284 def gen_cache(self, key, value=None, type=Cache): 

285 """ 

286 Generates cache for `forward` usage in inference accroding to arguments. 

287 The generated cache is an instance of `MultiHeadAttention.Cache` or an 

288 instance of `MultiHeadAttention.StaticCache`. 

289 

290 `Cache` or `StaticCache` is namedtuple with `k` and `v` as fields, 

291 and it stores tensors shaped `[batch_size, num_heads, length, embed_dim]` 

292 which are results of linear projection, reshape and transpose calculations 

293 in MultiHeadAttention. 

294  

295 If the generated cache is an instance of `Cache`, `k` and `v` fields 

296 reserve intermediate result tensors of previous positions, and the tensors 

297 are incremental among decoding steps, which mostly are used for decoder 

298 decoder self attention. 

299  

300 If the generated cache is an instance of `StaticCache`, `k` and `v` fields 

301 would be used as calculated result tensors on keys an values in `forward`, 

302 and the tensors keep unchanged among decoding steps, which are mostly used 

303 for decoder-encoder cross attention. 

304 

305 The cache is generated as follows: 

306 

307 1. If `type` is `StaticCache`, apply `compute_kv(key, value)` and use the 

308 results to create an instance of `StaticCache`. 

309  

310 2. If `type` is `Cache` and `value` is None, generate empty tensors shaped 

311 `[batch_size, num_heads, 0, embed_dim // num_heads]` and use the results 

312 to create an instance of `Cache`, where `batch_size` is from the first 

313 dimension of `key`. 

314 

315 3. If `type` is `Cache` and `value` is not None, use `key`, `value` to create 

316 an instance of `Cache`. 

317 

318 Parameters: 

319 key (Tensor): The keys for multi-head attention. It is 

320 a tensor with shape `[batch_size, key_length, kdim]`. The 

321 data type should be float32 or float64. If `value` is None, 

322 it is only for batch size and data type reference. 

323 value (Tensor, optional): The values for multi-head attention. It 

324 is a tensor with shape `[batch_size, value_length, vdim]`. 

325 The data type should be float32 or float64. If None, `key` is only 

326 for batch size reference. Default None. 

327 type (type): It should be `MultiHeadAttention.StaticCache` or 

328 `MultiHeadAttention.Cache` to indicate the cache type to generate. 

329  

330 Returns: 

331 namedtuple: an instance of `Cache` or `StaticCache` accordingly. 

332 """ 

333 if type == MultiHeadAttention.StaticCache: # static_kv 

334 k, v = self.compute_kv(key, value) 

335 return self.StaticCache(k, v) 

336 elif value is None: # incremental_state 

337 k = layers.fill_constant_batch_size_like( 

338 input=key, 

339 shape=[-1, self.num_heads, 0, self.head_dim], 

340 dtype=key.dtype, 

341 value=0) 

342 v = layers.fill_constant_batch_size_like( 

343 input=key, 

344 shape=[-1, self.num_heads, 0, self.head_dim], 

345 dtype=key.dtype, 

346 value=0) 

347 return self.Cache(k, v) 

348 else: 

349 # incremental_state with initial value, mainly for usage like UniLM 

350 return self.Cache(key, value)

352 def forward(self, query, key=None, value=None, attn_mask=None, cache=None): 

353 r""" 

354 Applies multi-head attention to map queries and a set of key-value pairs 

355 to outputs. 

356 

357 Parameters: 

358 query (Tensor): The queries for multi-head attention. It is a 

359 tensor with shape `[batch_size, query_length, embed_dim]`. The 

360 data type should be float32 or float64. 

361 key (Tensor, optional): The keys for multi-head attention. It is 

362 a tensor with shape `[batch_size, key_length, kdim]`. The 

363 data type should be float32 or float64. If None, use `query` as 

364 `key`. Default None. 

365 value (Tensor, optional): The values for multi-head attention. It 

366 is a tensor with shape `[batch_size, value_length, vdim]`. 

367 The data type should be float32 or float64. If None, use `query` as 

368 `value`. Default None. 

369 attn_mask (Tensor, optional): A tensor used in multi-head attention 

370 to prevents attention to some unwanted positions, usually the 

371 paddings or the subsequent positions. It is a tensor with shape 

372 broadcasted to `[batch_size, n_head, sequence_length, sequence_length]`. 

373 When the data type is bool, the unwanted positions have `False`  

374 values and the others have `True` values. When the data type is  

375 int, the unwanted positions have 0 values and the others have 1  

376 values. When the data type is float, the unwanted positions have  

377 `-INF` values and the others have 0 values. It can be None when  

378 nothing wanted or needed to be prevented attention to. Default None. 

379 cache (MultiHeadAttention.Cache|MultiHeadAttention.StaticCache, optional): 

380 It is a namedtuple with `k` and `v` as fields, and stores tensors 

381 shaped `[batch_size, num_heads, length, embed_dim]` which are results 

382 of linear projection, reshape and transpose calculations in 

383 MultiHeadAttention. If it is an instance of `Cache`, `k` and `v` 

384 fields reserve intermediate results of previous positions, which 

385 mostly used for decoder self attention. If it is an instance of 

386 `StaticCache`, `key` and `value` args would be ignored, `k` and 

387 `v` fields would be used as calculated results on `key` and 

388 `value`, which mostly used for decoder-encoder cross attention. 

389 It is only used for inference and should be None for training. 

390 Default None. 

391 

392 Returns: 

393 Tensor|tuple: It is a tensor that has the same shape and data type \ 

394 as `query`, representing attention output. Or a tuple if \ 

395 `need_weights` is True or `cache` is not None. If `need_weights` \ 

396 is True, except for attention output, the tuple also includes \ 

397 the attention weights tensor shaped `[batch_size, num_heads, query_length, key_length]`. \ 

398 If `cache` is not None, the tuple then includes the new cache \ 

399 having the same type as `cache`, and if it is `StaticCache`, it \ 

400 is same as the input `cache`, if it is `Cache`, the new cache \ 

401 reserves tensors concatanating raw tensors with intermediate \ 

402 results of current query. 

403 """ 

404 key = query if key is None else key 

405 value = query if value is None else value 

406 # compute q ,k ,v 

407 if cache is None: 

408 q, k, v = self._prepare_qkv(query, key, value, cache) 

409 else: 

410 q, k, v, cache = self._prepare_qkv(query, key, value, cache) 

411 

412 # scale dot product attention 

413 product = paddle.matmul(x=q * (self.head_dim**-0.5), 

414 y=k, 

415 transpose_y=True) 

416 if attn_mask is not None: 

417 # Support bool or int mask 

418 attn_mask = _convert_attention_mask(attn_mask, product.dtype) 

419 product = product + attn_mask 

420 weights = F.softmax(product) 

421 if self.dropout: 

422 weights = F.dropout(weights, 

423 self.dropout, 

424 training=self.training, 

425 mode="upscale_in_train") 

426 

427 out = tensor.matmul(weights, v) 

428 

429 # combine heads 

430 out = tensor.transpose(out, perm=[0, 2, 1, 3]) 

431 out = tensor.reshape(x=out, shape=[0, 0, out.shape[2] * out.shape[3]]) 

432 

433 # project to output 

434 out = self.out_proj(out) 

435 

436 outs = [out] 

437 if self.need_weights: 

438 outs.append(weights) 

439 if cache is not None: 

440 outs.append(cache) 

441 return out if len(outs) == 1 else tuple(outs)

444class TransformerEncoderLayer(Layer): 

445 """ 

446 TransformerEncoderLayer is composed of two sub-layers which are self (multi-head) 

447 attention and feedforward network. Before and after each sub-layer, pre-process 

448 and post-precess would be applied on the input and output accordingly. If 

449 `normalize_before` is True, pre-process is layer normalization and post-precess 

450 includes dropout, residual connection. Otherwise, no pre-process and post-precess 

451 includes dropout, residual connection, layer normalization. 

452 

453 Parameters: 

454 d_model (int): The expected feature size in the input and output. 

455 nhead (int): The number of heads in multi-head attention(MHA). 

456 dim_feedforward (int): The hidden layer size in the feedforward network(FFN). 

457 dropout (float, optional): The dropout probability used in pre-process 

458 and post-precess of MHA and FFN sub-layer. Default 0.1 

459 activation (str, optional): The activation function in the feedforward 

460 network. Default relu. 

461 attn_dropout (float, optional): The dropout probability used 

462 in MHA to drop some attention target. If None, use the value of 

463 `dropout`. Default None 

464 act_dropout (float, optional): The dropout probability used after FFN 

465 activition. If None, use the value of `dropout`. Default None 

466 normalize_before (bool, optional): Indicate whether to put layer normalization 

467 into preprocessing of MHA and FFN sub-layers. If True, pre-process is layer 

468 normalization and post-precess includes dropout, residual connection. 

469 Otherwise, no pre-process and post-precess includes dropout, residual 

470 connection, layer normalization. Default False 

471 weight_attr(ParamAttr|list|tuple, optional): To specify the weight parameter property. 

472 If it is a list/tuple, `weight_attr[0]` would be used as `weight_attr` for 

473 MHA, and `weight_attr[1]` would be used as `weight_attr` for linear in FFN. 

474 Otherwise, MHA and FFN both use it as `weight_attr` to create parameters. 

475 Default: None, which means the default weight parameter property is used. 

476 See usage for details in :code:`ParamAttr` .  

477 bias_attr (ParamAttr|list|tuple|bool, optional): To specify the bias parameter property. 

478 If it is a list/tuple, `bias_attr[0]` would be used as `bias_attr` for 

479 MHA, and `bias_attr[1]` would be used as `bias_attr` for linear in FFN. 

480 Otherwise, MHA and FFN both use it as `bias_attr` to create parameters. 

481 The `False` value means the corresponding layer would not have trainable 

482 bias parameter. See usage for details in :code:`ParamAttr` . Default: None, 

483 which means the default bias parameter property is used. 

484

553 def forward(self, src, src_mask=None, cache=None): 

554 r""" 

555 Applies a Transformer encoder layer on the input. 

556 

557 Parameters: 

558 src (Tensor): The input of Transformer encoder layer. It is 

559 a tensor with shape `[batch_size, sequence_length, d_model]`. 

560 The data type should be float32 or float64. 

561 src_mask (Tensor, optional): A tensor used in multi-head attention 

562 to prevents attention to some unwanted positions, usually the 

563 paddings or the subsequent positions. It is a tensor with shape 

564 broadcasted to `[batch_size, n_head, sequence_length, sequence_length]`. 

565 When the data type is bool, the unwanted positions have `False`  

566 values and the others have `True` values. When the data type is  

567 int, the unwanted positions have 0 values and the others have 1  

568 values. When the data type is float, the unwanted positions have  

569 `-INF` values and the others have 0 values. It can be None when  

570 nothing wanted or needed to be prevented attention to. Default None. 

571 cache (Tensor, optional): It is an instance of `MultiHeadAttention.Cache`. 

572 See `TransformerEncoderLayer.gen_cache` for more details. It is 

573 only used for inference and should be None for training. Default 

574 None. 

575 

576 Returns: 

577 Tensor|tuple: It is a tensor that has the same shape and data type \ 

578 as `enc_input`, representing the output of Transformer encoder \ 

579 layer. Or a tuple if `cache` is not None, except for encoder \ 

580 layer output, the tuple includes the new cache which is same \ 

581 as input `cache` argument but `incremental_cache` has an \ 

582 incremental length. See `MultiHeadAttention.gen_cache` and \ 

583 `MultiHeadAttention.forward` for more details. 

584 """ 

585 src_mask = _convert_attention_mask(src_mask, src.dtype) 

586 

587 residual = src 

588 if self.normalize_before: 

589 src = self.norm1(src) 

590 # Add cache for encoder for the usage like UniLM 

591 if cache is None: 

592 src = self.self_attn(src, src, src, src_mask) 

593 else: 

594 src, incremental_cache = self.self_attn(src, src, src, src_mask, 

595 cache) 

596 

597 src = residual + self.dropout1(src) 

598 if not self.normalize_before: 

599 src = self.norm1(src) 

600 

601 residual = src 

602 if self.normalize_before: 

603 src = self.norm2(src) 

604 src = self.linear2(self.dropout(self.activation(self.linear1(src)))) 

605 src = residual + self.dropout2(src) 

606 if not self.normalize_before: 

607 src = self.norm2(src) 

608 return src if cache is None else (src, incremental_cache)

628 type=self.self_attn.Cache) 

629 return incremental_cache

680 to prevents attention to some unwanted positions, usually the

683 When the data type is bool, the unwanted positions have `False`  

684 values and the others have `True` values. When the data type is  

685 int, the unwanted positions have 0 values and the others have 1  

686 values. When the data type is float, the unwanted positions have  

687 `-INF` values and the others have 0 values. It can be None when  

688 nothing wanted or needed to be prevented attention to. Default None.

718
