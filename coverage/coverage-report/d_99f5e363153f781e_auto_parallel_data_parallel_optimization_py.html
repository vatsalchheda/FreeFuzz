40@register_pass("auto_parallel_data_parallel_optimization") 

41class DataParallelOptimizationPass(PassBase): 

42 """ 

43 Apply Optimizations that specialized for data parallelism in Auto Parallel. 

44 1. prune grad scaling  

45 2. overlap comm and calc 

46 3. fuse allreduce 

47 """

121 def _analyze_program(self): 

122 """ 

123 build two maps 

124 {param_grad_name: data_parallel_group} 

125 {pdata_parallel_group: aram_grad_name} 

126 """ 

127 

128 block = default_main_program().global_block() 

129 ops = block.ops 

130 scaled_grads = [] 

131 

132 for op in ops: 

133 

134 if is_data_parallel_reduce_op(op): 

135 grad_name = op.output_arg_names[0] 

136 if grad_name in self._grad_name_to_group_map: 

137 continue 

138 assert op.has_attr( 

139 "ring_id" 

140 ), "Unexception: comm op [{}] has NOT ring id.".format(str(op)) 

141 group = ring_id_to_process_group(op.attr("ring_id")) 

142 

143 assert group is not None, "Unexception: data parallel group of [{}] from op [{}] is None".format( 

144 grad_name, str(op)) 

145 

146 self._grad_name_to_group_map[grad_name] = group 

147 

148 if group not in self._group_to_grad_name_map: 

149 self._group_to_grad_name_map[group] = [grad_name] 

150 else: 

151 self._group_to_grad_name_map[group].append(grad_name) 

152 

153 elif is_data_parallel_scale_op(op): 

154 grad_name = op.output_arg_names[0] 

155 scaled_grads.append(grad_name) 

156 

157 # TODO support multiple optimizers in on network in future. 

158 # here we assume that the optimizer is unique in network. 

159 elif is_optimize_op( 

160 op) and op.type in __rescale_grad_supported_opts__: 

161 self._support_rescale_grad = True 

162 

163 not_synchronized_grads = [] 

164 for grad_name in scaled_grads: 

165 if grad_name not in self._grad_name_to_group_map: 

166 not_synchronized_grads.append(grad_name) 

167 assert len( 

168 not_synchronized_grads 

169 ) == 0, "Unexception: gradients [{}] is scaled BUT NOT synchronized.".format( 

170 not_synchronized_grads)

172 def is_data_parallel_applied(self): 

173 return len(self._group_to_grad_name_map) > 0

175 def _could_be_prune(self): 

176 

177 return self.dist_context.gradient_scale and ( 

178 self._support_rescale_grad or self._all_dp_groups_same_degree())

180 def _all_dp_groups_same_degree(self): 

181 return len( 

182 set([ 

183 len(group.ranks) 

184 for group in self._group_to_grad_name_map.keys() 

185 ])) == 1

187 def _scale_backward_initial_grad(self): 

188 

189 block = default_main_program().global_block() 

190 dp_degree = len(list(self._group_to_grad_name_map.keys())[0].ranks) 

191 

192 for idx, op in reversed(list(enumerate(block.ops))): 

193 if is_loss_grad_op(op): 

194 assert op.type == 'fill_constant', \ 

195 "loss_grad_op must be fill_constant op, " \ 

196 "but this op is {}".format(op.type) 

197 assert op.has_attr('value') 

198 loss_scale = float(op.attr('value')) 

199 loss_scale = loss_scale / dp_degree 

200 op._set_attr('value', loss_scale) 

201 break

212 def _update_opt_rescale_grad(self): 

213 

214 block = default_main_program().global_block() 

215 scaled_grads = set() 

216 

217 for idx, op in reversed(list(enumerate(block.ops))): 

218 if is_optimize_op( 

219 op) and op.type in __rescale_grad_supported_opts__: 

220 assert op.has_attr( 

221 'rescale_grad' 

222 ), "Unexception: op [{}] is supported to have [rescale_grad] attribute.".format( 

223 str(op)) 

224 assert len( 

225 op.input("Grad") 

226 ) == 1, "Unexception: op [{}] is supported to have only one input grad var.".format( 

227 str(op)) 

228 

229 grad_name = op.input("Grad")[0] 

230 dp_degree = len( 

231 list(self._grad_name_to_group_map[grad_name].ranks)) 

232 scaled_grads.add(grad_name) 

233 

234 rescale_grad = float(op.attr('rescale_grad')) / dp_degree 

235 op._set_attr('rescale_grad', rescale_grad) 

236 

237 assert scaled_grads == set(self._grad_name_to_group_map.keys( 

238 )), "Unexception: gradients [{}] are unscaled.".format( 

239 set(self._grad_name_to_group_map.keys()) - scaled_grads)

241 def _could_be_overlap(self): 

242 # NOTE current different nccl comm will use different cuda stream 

243 # so if there too many dp group there will be too many stream need to be 

244 # created and sync. 

245 # revise here when framework support custom stream in static mode. 

246 num_dp_comm_stream = len(set(self._group_to_grad_name_map.keys())) 

247 if num_dp_comm_stream > __max_stream_num_allow__: 

248 return False 

249 if self.use_sharding: 

250 return False 

251 return True

255 # InterpreterCore has a different logic for overlapping

280 def _calc_wait_comms(self): 

281 

282 block = default_main_program().global_block() 

283 ops = block.ops 

284 

285 # NOTE the naive overlap implement in static hybird parallel only sync comm stream 

286 # at the end of Backward phase, based on a strong constraint that 

287 # all communicating gradient would NOT be used after communication in Backward phase. 

288 # BUT this constraint will fail for scenario like Weight-Sharing and Higher-Order Differentiation, 

289 # where gradient will be involved in other calculation between data-parallel allreduce kernel submmited 

290 # into comm streams and the synchronization of comm stream at the end of Backward phase. 

291 # synchronization of comm stream should add according to the usage of communicating gradients 

292 # to support Overlapping for Weight-Sharing and Higher-Order Differentiation. 

293 

294 ring_id_to_un_sync_grad_map = {} 

295 op_idx_to_sync_ring_id_map = {} 

296 for group in self._group_to_grad_name_map.keys(): 

297 ring_id_to_un_sync_grad_map[group.id] = [] 

298 

299 # analyze the where need to sync 

300 for i, op in enumerate(ops): 

301 if is_data_parallel_reduce_op(op): 

302 ring_id = op.attr("ring_id") 

303 grad_name = op.output_arg_names[0] 

304 ring_id_to_un_sync_grad_map[ring_id].append(grad_name) 

305 elif is_data_parallel_scale_op(op): 

306 continue 

307 # other ops that might use communicating grad 

308 else: 

309 for input_var_name in op.input_arg_names: 

310 for ring_id, unsync_grad_names in ring_id_to_un_sync_grad_map.items( 

311 ): 

312 if input_var_name in unsync_grad_names: 

313 # need to sync before op_i 

314 if i in op_idx_to_sync_ring_id_map: 

315 op_idx_to_sync_ring_id_map[i].append(ring_id) 

316 else: 

317 op_idx_to_sync_ring_id_map[i] = [ring_id] 

318 # all grads in this comm stream are synced 

319 ring_id_to_un_sync_grad_map[ring_id] = [] 

320 

321 # insert synchronization 

322 indices = list(op_idx_to_sync_ring_id_map.keys()) 

323 # TODO the synchronization could be optimized 

324 # we should record the event of a gradient is communicating and 

325 # only wait for that event to be completed. 

326 # BUT paddle static currently not support op api for event record only, so 

327 # here we try to wait for all kernel in that comm stream to be finish which is not that optimized. 

328 for i in sorted(indices, reverse=True): 

329 for ring_id in op_idx_to_sync_ring_id_map[i]: 

330 

331 block._insert_op_without_sync(i, 

332 type='c_wait_comm', 

333 inputs={'X': []}, 

334 outputs={'Out': []}, 

335 attrs={ 

336 'op_role': OpRole.Backward, 

337 'ring_id': ring_id 

338 })

349 def _group_grads(self): 

350 """ 

351 conditions for gradients to be grouped: 

352 1. group size < max_fuse_numel 

353 2. same dp group  

354 3. same dtype 

355 4. dependency: grad would NOT be used by other ops within group segment  

356 

357 gradients inside same group would be fuse into one coalesce tensor 

358 """ 

359 

360 block = default_main_program().global_block() 

361 ops = block.ops 

362 

363 # group individual grad vars 

364 # TODO consider fuse gradient for sharding reduce 

365 # TODO let user to set fuse_grad_size 

366 # emb = 50000 * h, ffn = 8 * h * h, mha = 4 * h * h 

367 h = 2048 

368 ffn_numel = 2 * (4 * h) * h 

369 mha_numel = 3 * h * h + h * h 

370 max_fuse_numel = ffn_numel + mha_numel 

371 grad_groups = [] 

372 cur_group = GradientsGroup(ops, max_fuse_numel) 

373 grouped_grad_names = set() 

374 

375 def collect_group(cur_group, grad_var, ring_id, i): 

376 if len(cur_group.gradients) == 0: 

377 cur_group = None 

378 elif len(cur_group.gradients) == 1: 

379 grouped_grad_names.remove(cur_group.gradients[0].name) 

380 else: 

381 cur_group.finalize() 

382 grad_groups.append(cur_group) 

383 

384 new_group = GradientsGroup(ops, max_fuse_numel) 

385 if grad_var: 

386 new_group.add(grad_var, ring_id, i) 

387 grouped_grad_names.add(grad_var.name) 

388 return new_group 

389 

390 def op_depend_on_group(op, group): 

391 vars_ = set(op.input_arg_names + op.output_arg_names) 

392 grad_names = set([grad.name for grad in group.gradients]) 

393 return len(vars_.intersection(grad_names)) > 0 

394 

395 for i, op in enumerate(ops): 

396 if is_data_parallel_reduce_op(op): 

397 ring_id = op.attr("ring_id") 

398 grad_name = op.output_arg_names[0] 

399 grad_var = block.var(grad_name) 

400 grad_numel = numel(grad_var) 

401 

402 if cur_group.acceptable(grad_var, ring_id): 

403 assert grad_name not in grouped_grad_names 

404 grouped_grad_names.add(grad_name) 

405 cur_group.add(grad_var, ring_id, i) 

406 else: 

407 cur_group = collect_group(cur_group, grad_var, ring_id, i) 

408 else: 

409 if op_depend_on_group(op, cur_group): 

410 cur_group = collect_group(cur_group, None, None, None) 

411 

412 # collect last group 

413 collect_group(cur_group, None, None, None) 

414 

415 return grad_groups

417 def _update_program(self, grad_groups): 

418 

419 block = default_main_program().global_block() 

420 

421 remove_op_types = ['scale', 'c_allreduce_sum', 'c_wait_compute'] 

422 

423 for i, group in enumerate(grad_groups[::-1]): 

424 

425 # create coalecse tensor 

426 group.coalesce_var = block.create_var(name=unique_name.generate( 

427 'coalecse_grad_{}'.format(i)), 

428 dtype=group.dtype, 

429 persistable=False, 

430 stop_gradient=True) 

431 

432 # update allreduce & scale op 

433 if group.scale_op_idx != -1: 

434 scale_op = block.ops[group.scale_op_idx] 

435 assert scale_op.type == 'scale', "should found scale op but found {}".format( 

436 str(scale_op)) 

437 scale_op._rename_input(scale_op.input_arg_names[0], 

438 group.coalesce_var.name) 

439 scale_op._rename_output(scale_op.output_arg_names[0], 

440 group.coalesce_var.name) 

441 

442 allreduce_op = block.ops[group.allreduce_op_idx] 

443 assert allreduce_op.type == 'c_allreduce_sum', "should found c_allreduce_sum op but found {}".format( 

444 str(allreduce_op)) 

445 allreduce_op._rename_input(allreduce_op.input_arg_names[0], 

446 group.coalesce_var.name) 

447 allreduce_op._rename_output(allreduce_op.output_arg_names[0], 

448 group.coalesce_var.name) 

449 

450 # remvoe un-used op 

451 remove_op_indices = group.remove_wait_op_indices + group.remove_allreduce_op_indices + group.remove_scale_op_indices 

452 for idx in sorted(remove_op_indices, reverse=True): 

453 assert block.ops[ 

454 idx].type in remove_op_types, "Unexception: try to remove op {}".format( 

455 str(op)) 

456 block._remove_op(idx) 

457 

458 # insert coalecse op 

459 concated_shapes = [] 

460 concated_ranks = [] 

461 for grad_ in group.gradients: 

462 shape = grad_.shape 

463 concated_shapes.extend(shape) 

464 concated_ranks.append(len(shape)) 

465 

466 grad_names = [grad.name for grad in group.gradients] 

467 block._insert_op_without_sync(group.coalesce_op_idx, 

468 type="coalesce_tensor", 

469 inputs={"Input": grad_names}, 

470 outputs={ 

471 "Output": grad_names, 

472 "FusedOutput": group.coalesce_var 

473 }, 

474 attrs={ 

475 "copy_data": False, 

476 "use_align": True, 

477 "dtype": group.dtype, 

478 "concated_shapes": 

479 concated_shapes, 

480 "concated_ranks": concated_ranks, 

481 OP_ROLE_KEY: OpRole.Backward 

482 }) 

483 

484 block._sync_with_cpp()

521class GradientsGroup(object): 

522 

523 def __init__(self, ops, max_group_size): 

524 self.max_group_size = max_group_size 

525 self.ops = ops 

526 

527 self.gradients = [] 

528 self.numel = 0 

529 self.dtype = None 

530 self.ring_id = None 

531 self.coalesce_var = None 

532 self.coalesce_op_idx = -1 

533 self.allreduce_op_idx = -1 

534 self.scale_op_idx = -1 

535 self.remove_wait_op_indices = [] 

536 self.remove_allreduce_op_indices = [] 

537 self.remove_scale_op_indices = [] 

538 

539 def acceptable(self, grad_var, ring_id): 

540 if len(self.gradients) == 0: 

541 return True 

542 if ring_id != self.ring_id: 

543 return False 

544 if numel(grad_var) + self.numel > self.max_group_size: 

545 return False 

546 if grad_var.dtype != self.dtype: 

547 return False 

548 

549 return True 

550 

551 def add(self, grad_var, ring_id, i): 

552 self.gradients.append(grad_var) 

553 self.ring_id = ring_id 

554 self.dtype = grad_var.dtype 

555 self.numel += numel(grad_var) 

556 

557 # remove auxiliary ops in non-fuse dp allreduce 

558 self.remove_allreduce_op_indices.append(i) 

559 

560 # NOTE this pass rely on the original synchronization add in previous passes 

561 # (same stream or calc_wait_comm & comm_wait_calc) 

562 # to guarantee the correctness of comm_calc execution order. 

563 # so the calc_wait_comm should be keep. 

564 grad_op_idx = i - 1 

565 if i > 0 and self.ops[i - 1].type == 'c_wait_compute': 

566 self.remove_wait_op_indices.append(i - 1) 

567 grad_op_idx -= 1 

568 if i + 1 < len(self.ops) and is_data_parallel_scale_op(self.ops[i - 1]): 

569 self.remove_scale_op_indices.append(i + 1) 

570 

571 if len(self.gradients) == 1: 

572 # TODO Remove this is a temporary hack for Tensor Parallel. the logic 

573 # for find grad_op should be more general. 

574 if self.ops[grad_op_idx].type == "c_allreduce_sum": 

575 grad_op_idx -= 1 

576 

577 grad_op = self.ops[grad_op_idx] 

578 assert grad_var.name in grad_op.output_arg_names, "grad [{}] should be output of {}".format( 

579 grad_var.name, str(grad_op)) 

580 self.coalesce_op_idx = grad_op_idx 

581 

582 def finalize(self): 

583 self.allreduce_op_idx = self.remove_allreduce_op_indices.pop() 

584 if len(self.remove_wait_op_indices) > 1: 

585 self.remove_wait_op_indices.pop() 

586 if len(self.remove_scale_op_indices) > 1: 

587 self.scale_op_idx = self.remove_scale_op_indices.pop()
