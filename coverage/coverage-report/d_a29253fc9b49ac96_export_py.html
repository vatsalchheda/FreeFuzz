21def export(layer, path, input_spec=None, opset_version=9, **configs): 

22 """ 

23 Export Layer to ONNX format, which can use for inference via onnxruntime or other backends. 

24 For more details, Please refer to `paddle2onnx <https://github.com/PaddlePaddle/paddle2onnx>`_ . 

25 

26 Args: 

27 layer (Layer): The Layer to be exported. 

28 path (str): The path prefix to export model. The format is ``dirname/file_prefix`` or ``file_prefix`` , 

29 and the exported ONNX file suffix is ``.onnx`` . 

30 input_spec (list[InputSpec|Tensor], optional): Describes the input of the exported model's forward 

31 method, which can be described by InputSpec or example Tensor. If None, all input variables of 

32 the original Layer's forward method would be the inputs of the exported ``ONNX`` model. Default: None. 

33 opset_version(int, optional): Opset version of exported ONNX model. 

34 Now, stable supported opset version include 9, 10, 11. Default: 9. 

35 **configs (dict, optional): Other export configuration options for compatibility. We do not 

36 recommend using these configurations, they may be removed in the future. If not necessary, 

37 DO NOT use them. Default None. 

38 The following options are currently supported: 

39 (1) output_spec (list[Tensor]): Selects the output targets of the exported model. 

40 By default, all return variables of original Layer's forward method are kept as the 

41 output of the exported model. If the provided ``output_spec`` list is not all output variables, 

42 the exported model will be pruned according to the given ``output_spec`` list. 

43 Returns: 

44 None 

45 Examples: 

46 .. code-block:: python 

47 

48 import paddle 

49 

50 class LinearNet(paddle.nn.Layer): 

51 def __init__(self): 

52 super(LinearNet, self).__init__() 

53 self._linear = paddle.nn.Linear(128, 10) 

54 

55 def forward(self, x): 

56 return self._linear(x) 

57 

58 # Export model with 'InputSpec' to support dynamic input shape. 

59 def export_linear_net(): 

60 model = LinearNet() 

61 x_spec = paddle.static.InputSpec(shape=[None, 128], dtype='float32') 

62 paddle.onnx.export(model, 'linear_net', input_spec=[x_spec]) 

63 

64 export_linear_net() 

65 

66 class Logic(paddle.nn.Layer): 

67 def __init__(self): 

68 super(Logic, self).__init__() 

69 

70 def forward(self, x, y, z): 

71 if z: 

72 return x 

73 else: 

74 return y 

75 

76 # Export model with 'Tensor' to support pruned model by set 'output_spec'. 

77 def export_logic(): 

78 model = Logic() 

79 x = paddle.to_tensor([1]) 

80 y = paddle.to_tensor([2]) 

81 # Static and run model. 

82 paddle.jit.to_static(model) 

83 out = model(x, y, z=True) 

84 paddle.onnx.export(model, 'pruned', input_spec=[x], output_spec=[out]) 

85 

86 export_logic() 

87 """ 

88 

89 p2o = try_import('paddle2onnx') 

90 

91 file_prefix = os.path.basename(path) 

92 if file_prefix == "": 

93 raise ValueError( 

94 "The input path MUST be format of dirname/file_prefix " 

95 "[dirname\\file_prefix in Windows system], but " 

96 "the file_prefix is empty in received path: {}".format(path) 

97 ) 

98 save_file = path + '.onnx' 

99 

100 p2o.dygraph2onnx( 

101 layer, 

102 save_file, 

103 input_spec=input_spec, 

104 opset_version=opset_version, 

105 **configs 

106 )
