30def graph_reindex( 

31 x, 

32 neighbors, 

33 count, 

34 value_buffer=None, 

35 index_buffer=None, 

36 flag_buffer_hashtable=False, 

37 name=None, 

38): 

39 """ 

40 

41 Graph Reindex API. 

42 

43 This API is mainly used in Graph Learning domain, which should be used 

44 in conjunction with `graph_sample_neighbors` API. And the main purpose 

45 is to reindex the ids information of the input nodes, and return the  

46 corresponding graph edges after reindex. 

47 

48 Notes: 

49 The number in x should be unique, otherwise it would cause potential errors. 

50 Besides, we also support multi-edge-types neighbors reindexing. If we have different 

51 edge_type neighbors for x, we should concatenate all the neighbors and count of x. 

52 We will reindex all the nodes from 0. 

53 

54 Take input nodes x = [0, 1, 2] as an example.  

55 If we have neighbors = [8, 9, 0, 4, 7, 6, 7], and count = [2, 3, 2],  

56 then we know that the neighbors of 0 is [8, 9], the neighbors of 1 

57 is [0, 4, 7], and the neighbors of 2 is [6, 7]. 

58 

59 Args: 

60 x (Tensor): The input nodes which we sample neighbors for. The available 

61 data type is int32, int64. 

62 neighbors (Tensor): The neighbors of the input nodes `x`. The data type 

63 should be the same with `x`. 

64 count (Tensor): The neighbor count of the input nodes `x`. And the  

65 data type should be int32. 

66 value_buffer (Tensor, optional): Value buffer for hashtable. The data type should 

67 be int32, and should be filled with -1. Default is None. 

68 index_buffer (Tensor, optional): Index buffer for hashtable. The data type should 

69 be int32, and should be filled with -1. Default is None. 

70 flag_buffer_hashtable (bool, optional): Whether to use buffer for hashtable to speed up. 

71 Default is False. Only useful for gpu version currently. 

72 name (str, optional): Name for the operation (optional, default is None). 

73 For more information, please refer to :ref:`api_guide_Name`. 

74  

75 Returns: 

76 - reindex_src (Tensor), The source node index of graph edges after reindex. 

77 - reindex_dst (Tensor), The destination node index of graph edges after reindex. 

78 - out_nodes (Tensor), The index of unique input nodes and neighbors before reindex, 

79 where we put the input nodes `x` in the front, and put neighbor 

80 nodes in the back. 

81 

82 Examples: 

83 .. code-block:: python 

84 

85 import paddle 

86 

87 x = [0, 1, 2] 

88 neighbors_e1 = [8, 9, 0, 4, 7, 6, 7] 

89 count_e1 = [2, 3, 2] 

90 x = paddle.to_tensor(x, dtype="int64") 

91 neighbors_e1 = paddle.to_tensor(neighbors_e1, dtype="int64") 

92 count_e1 = paddle.to_tensor(count_e1, dtype="int32") 

93 

94 reindex_src, reindex_dst, out_nodes = \ 

95 paddle.incubate.graph_reindex(x, neighbors_e1, count_e1) 

96 # reindex_src: [3, 4, 0, 5, 6, 7, 6] 

97 # reindex_dst: [0, 0, 1, 1, 1, 2, 2] 

98 # out_nodes: [0, 1, 2, 8, 9, 4, 7, 6] 

99 

100 neighbors_e2 = [0, 2, 3, 5, 1] 

101 count_e2 = [1, 3, 1] 

102 neighbors_e2 = paddle.to_tensor(neighbors_e2, dtype="int64") 

103 count_e2 = paddle.to_tensor(count_e2, dtype="int32") 

104 

105 neighbors = paddle.concat([neighbors_e1, neighbors_e2]) 

106 count = paddle.concat([count_e1, count_e2]) 

107 reindex_src, reindex_dst, out_nodes = \ 

108 paddle.incubate.graph_reindex(x, neighbors, count) 

109 # reindex_src: [3, 4, 0, 5, 6, 7, 6, 0, 2, 8, 9, 1] 

110 # reindex_dst: [0, 0, 1, 1, 1, 2, 2, 0, 1, 1, 1, 2] 

111 # out_nodes: [0, 1, 2, 8, 9, 4, 7, 6, 3, 5] 

112 

113 """ 

114 if flag_buffer_hashtable: 

115 if value_buffer is None or index_buffer is None: 

116 raise ValueError( 

117 f"`value_buffer` and `index_buffer` should not" 

118 "be None if `flag_buffer_hashtable` is True." 

119 ) 

120 

121 if _non_static_mode(): 

122 reindex_src, reindex_dst, out_nodes = _legacy_C_ops.graph_reindex( 

123 x, 

124 neighbors, 

125 count, 

126 value_buffer, 

127 index_buffer, 

128 "flag_buffer_hashtable", 

129 flag_buffer_hashtable, 

130 ) 

131 return reindex_src, reindex_dst, out_nodes 

132 

133 check_variable_and_dtype(x, "X", ("int32", "int64"), "graph_reindex") 

134 check_variable_and_dtype( 

135 neighbors, "Neighbors", ("int32", "int64"), "graph_reindex" 

136 )

144 index_buffer, "HashTable_Index", ("int32"), "graph_reindex" 

145 ) 

146 

147 helper = LayerHelper("graph_reindex", **locals()) 

148 reindex_src = helper.create_variable_for_type_inference(dtype=x.dtype) 

149 reindex_dst = helper.create_variable_for_type_inference(dtype=x.dtype) 

150 out_nodes = helper.create_variable_for_type_inference(dtype=x.dtype) 

151 helper.append_op( 

152 type="graph_reindex", 

153 inputs={ 

154 "X": x, 

155 "Neighbors": neighbors, 

156 "Count": count, 

157 "HashTable_Value": value_buffer if flag_buffer_hashtable else None, 

158 "HashTable_Index": index_buffer if flag_buffer_hashtable else None, 

159 }, 

160 outputs={ 

161 "Reindex_Src": reindex_src, 

162 "Reindex_Dst": reindex_dst, 

163 "Out_Nodes": out_nodes, 

164 }, 

165 attrs={"flag_buffer_hashtable": flag_buffer_hashtable}, 

166 ) 

167 return reindex_src, reindex_dst, out_nodes
