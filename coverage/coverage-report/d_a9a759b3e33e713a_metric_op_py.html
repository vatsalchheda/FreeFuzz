40def accuracy(input, label, k=1, correct=None, total=None): 

41 """ 

42 

43 accuracy layer. 

44 Refer to the https://en.wikipedia.org/wiki/Precision_and_recall 

45 This function computes the accuracy using the input and label. 

46 If the correct label occurs in top k predictions, then correct will increment by one. 

47 

48 Note: 

49 the dtype of accuracy is determined by input. the input and label dtype can be different. 

50 

51 Args: 

52 input(Tensor): The input of accuracy layer, which is the predictions of network. A Tensor with type float32,float64. 

53 The shape is ``[sample_number, class_dim]`` . 

54 label(Tensor): The label of dataset. Tensor with type int32,int64. The shape is ``[sample_number, 1]`` . 

55 k(int, optional): The top k predictions for each class will be checked. Data type is int64 or int32. Default is 1. 

56 correct(Tensor, optional): The correct predictions count. A Tensor with type int64 or int32. Default is None. 

57 total(Tensor, optional): The total entries count. A tensor with type int64 or int32. Default is None. 

58 

59 Returns: 

60 Tensor, The correct rate. A Tensor with type float32. 

61 

62 Examples: 

63 .. code-block:: python 

64 

65 import numpy as np 

66 import paddle 

67 import paddle.static as static 

68 import paddle.nn.functional as F 

69 paddle.enable_static() 

70 data = static.data(name="input", shape=[-1, 32, 32], dtype="float32") 

71 label = static.data(name="label", shape=[-1,1], dtype="int") 

72 fc_out = static.nn.fc(x=data, size=10) 

73 predict = F.softmax(x=fc_out) 

74 result = static.accuracy(input=predict, label=label, k=5) 

75 place = paddle.CPUPlace() 

76 exe = static.Executor(place) 

77 exe.run(static.default_startup_program()) 

78 x = np.random.rand(3, 32, 32).astype("float32") 

79 y = np.array([[1],[0],[1]]) 

80 output= exe.run(feed={"input": x,"label": y}, 

81 fetch_list=[result[0]]) 

82 print(output) 

83 #[array([0.], dtype=float32)] 

84 

85 """ 

86 if _non_static_mode(): 

87 if correct is None: 

88 correct = _varbase_creator(dtype="int32") 

89 if total is None: 

90 total = _varbase_creator(dtype="int32") 

91 

92 _k = k.numpy().item(0) if isinstance(k, Variable) else k 

93 topk_out, topk_indices = _legacy_C_ops.top_k_v2( 

94 input, 'k', _k, 'sorted', False 

95 ) 

96 _acc, _, _ = _legacy_C_ops.accuracy( 

97 topk_out, topk_indices, label, correct, total 

98 ) 

99 return _acc 

100 

101 helper = LayerHelper("accuracy", **locals()) 

102 check_variable_and_dtype( 

103 input, 'input', ['float16', 'float32', 'float64'], 'accuracy' 

104 ) 

105 topk_out = helper.create_variable_for_type_inference(dtype=input.dtype) 

106 topk_indices = helper.create_variable_for_type_inference(dtype="int64") 

107 inputs = {"X": [input]} 

108 if isinstance(k, Variable): 

109 inputs['K'] = [k] 

110 else: 

111 attrs = {'k': k} 

112 attrs['sorted'] = False 

113 helper.append_op( 

114 type="top_k_v2", 

115 inputs=inputs, 

116 attrs=attrs, 

117 outputs={"Out": [topk_out], "Indices": [topk_indices]}, 

118 ) 

119 acc_out = helper.create_variable_for_type_inference(dtype="float32") 

120 if correct is None: 

121 correct = helper.create_variable_for_type_inference(dtype="int32") 

122 if total is None: 

123 total = helper.create_variable_for_type_inference(dtype="int32") 

124 helper.append_op( 

125 type="accuracy", 

126 inputs={"Out": [topk_out], "Indices": [topk_indices], "Label": [label]}, 

127 outputs={ 

128 "Accuracy": [acc_out], 

129 "Correct": [correct], 

130 "Total": [total], 

131 }, 

132 ) 

133 return acc_out

136def auc( 

137 input, 

138 label, 

139 curve='ROC', 

140 num_thresholds=2**12 - 1, 

141 topk=1, 

142 slide_steps=1, 

143 ins_tag_weight=None, 

144): 

145 """ 

146 **Area Under the Curve (AUC) Layer** 

147 

148 This implementation computes the AUC according to forward output and label. 

149 It is used very widely in binary classification evaluation. 

150 

151 Note: If input label contains values other than 0 and 1, it will be cast 

152 to `bool`. Find the relevant definitions `here <https://en.wikipedia.org\ 

153 /wiki/Receiver_operating_characteristic#Area_under_the_curve>`_. 

154 

155 There are two types of possible curves: 

156 

157 1. ROC: Receiver operating characteristic; 

158 2. PR: Precision Recall 

159 

160 Args: 

161 input(Tensor): A floating-point 2D Tensor, values are in the range 

162 [0, 1]. Each row is sorted in descending order. This 

163 input should be the output of topk. Typically, this 

164 Tensor indicates the probability of each label. 

165 A Tensor with type float32,float64. 

166 label(Tensor): A 2D int Tensor indicating the label of the training 

167 data. The height is batch size and width is always 1. 

168 A Tensor with type int32,int64. 

169 curve(str): Curve type, can be 'ROC' or 'PR'. Default 'ROC'. 

170 num_thresholds(int): The number of thresholds to use when discretizing 

171 the roc curve. Default 4095. 

172 topk(int): only topk number of prediction output will be used for auc. 

173 slide_steps: when calc batch auc, we can not only use step currently but the previous steps can be used. slide_steps=1 means use the current step, slide_steps=3 means use current step and the previous second steps, slide_steps=0 use all of the steps. 

174 ins_tag_weight(Tensor): A 2D int Tensor indicating the data's tag weight, 1 means real data, 0 means fake data. Default None, and it will be assigned to a tensor of value 1.  

175 A Tensor with type float32,float64. 

176 

177 Returns: 

178 Tensor: A tuple representing the current AUC. 

179 The return tuple is auc_out, batch_auc_out, [ 

180 batch_stat_pos, batch_stat_neg, stat_pos, stat_neg ] 

181 Data type is Tensor, supporting float32, float64. 

182 

183 Examples: 

184 .. code-block:: python 

185 

186 import paddle 

187 import numpy as np 

188 paddle.enable_static() 

189 

190 data = paddle.static.data(name="input", shape=[-1, 32,32], dtype="float32") 

191 label = paddle.static.data(name="label", shape=[-1], dtype="int") 

192 fc_out = paddle.static.nn.fc(x=data, size=2) 

193 predict = paddle.nn.functional.softmax(x=fc_out) 

194 result=paddle.static.auc(input=predict, label=label) 

195 

196 place = paddle.CPUPlace() 

197 exe = paddle.static.Executor(place) 

198 

199 exe.run(paddle.static.default_startup_program()) 

200 x = np.random.rand(3,32,32).astype("float32") 

201 y = np.array([1,0,1]) 

202 output= exe.run(feed={"input": x,"label": y}, 

203 fetch_list=[result[0]]) 

204 print(output) 

205  

206 #you can learn the usage of ins_tag_weight by the following code. 

207 ''' 

208 import paddle 

209 import numpy as np 

210 paddle.enable_static() 

211 

212 data = paddle.static.data(name="input", shape=[-1, 32,32], dtype="float32") 

213 label = paddle.static.data(name="label", shape=[-1], dtype="int") 

214 ins_tag_weight = paddle.static.data(name='ins_tag', shape=[-1,16], lod_level=0, dtype='float64') 

215 fc_out = paddle.static.nn.fc(x=data, size=2) 

216 predict = paddle.nn.functional.softmax(x=fc_out) 

217 result=paddle.static.auc(input=predict, label=label, ins_tag_weight=ins_tag_weight) 

218 

219 place = paddle.CPUPlace() 

220 exe = paddle.static.Executor(place) 

221 

222 exe.run(paddle.static.default_startup_program()) 

223 x = np.random.rand(3,32,32).astype("float32") 

224 y = np.array([1,0,1]) 

225 z = np.array([1,0,1]) 

226 output= exe.run(feed={"input": x,"label": y, "ins_tag_weight":z}, 

227 fetch_list=[result[0]]) 

228 print(output) 

229 ''' 

230 

231 """ 

232 helper = LayerHelper("auc", **locals()) 

233 

234 if ins_tag_weight is None: 

235 ins_tag_weight = tensor.fill_constant( 

236 shape=[1, 1], dtype="float32", value=1.0 

237 ) 

238 check_variable_and_dtype(input, 'input', ['float32', 'float64'], 'auc') 

239 check_variable_and_dtype(label, 'label', ['int32', 'int64'], 'auc') 

240 check_variable_and_dtype( 

241 ins_tag_weight, 'ins_tag_weight', ['float32', 'float64'], 'auc' 

242 ) 

243 auc_out = helper.create_variable_for_type_inference(dtype="float64") 

244 batch_auc_out = helper.create_variable_for_type_inference(dtype="float64") 

245 # make tp, tn, fp, fn persistable, so that can accumulate all batches. 

246 

247 # for batch auc 

248 # we create slide_step+1 buckets, the first slide_steps buckets store 

249 # historical batch-level values, and the last bucket stores the sum values of 

250 # previous slide_step buckets. 

251 # The index of bucket that the newest batch will use is determined by batch_id mod slide_steps, 

252 # and batch_id is store in the last posision of following variable 

253 batch_stat_pos = helper.create_global_variable( 

254 persistable=True, 

255 dtype='int64', 

256 shape=[(1 + slide_steps) * (num_thresholds + 1) + 1], 

257 ) 

258 batch_stat_neg = helper.create_global_variable( 

259 persistable=True, 

260 dtype='int64', 

261 shape=[(1 + slide_steps) * (num_thresholds + 1) + 1], 

262 ) 

263 

264 # for global auc 

265 # Needn't maintain the batch id 

266 stat_pos = helper.create_global_variable( 

267 persistable=True, dtype='int64', shape=[1, num_thresholds + 1] 

268 ) 

269 stat_neg = helper.create_global_variable( 

270 persistable=True, dtype='int64', shape=[1, num_thresholds + 1] 

271 ) 

272 

273 for var in [batch_stat_pos, batch_stat_neg, stat_pos, stat_neg]: 

274 helper.set_variable_initializer( 

275 var, Constant(value=0.0, force_cpu=False) 

276 ) 

277 

278 # "InsTagWeight": [ins_tag_weight] 

279 # Batch AUC 

280 helper.append_op( 

281 type="auc", 

282 inputs={ 

283 "Predict": [input], 

284 "Label": [label], 

285 "StatPos": [batch_stat_pos], 

286 "StatNeg": [batch_stat_neg], 

287 }, 

288 attrs={ 

289 "curve": curve, 

290 "num_thresholds": num_thresholds, 

291 "slide_steps": slide_steps, 

292 }, 

293 outputs={ 

294 "AUC": [batch_auc_out], 

295 "StatPosOut": [batch_stat_pos], 

296 "StatNegOut": [batch_stat_neg], 

297 }, 

298 ) 

299 # Global AUC 

300 helper.append_op( 

301 type="auc", 

302 inputs={ 

303 "Predict": [input], 

304 "Label": [label], 

305 "StatPos": [stat_pos], 

306 "StatNeg": [stat_neg], 

307 }, 

308 attrs={ 

309 "curve": curve, 

310 "num_thresholds": num_thresholds, 

311 "slide_steps": 0, 

312 }, 

313 outputs={ 

314 "AUC": [auc_out], 

315 "StatPosOut": [stat_pos], 

316 "StatNegOut": [stat_neg], 

317 }, 

318 ) 

319 return ( 

320 auc_out, 

321 batch_auc_out, 

322 [batch_stat_pos, batch_stat_neg, stat_pos, stat_neg], 

323 )
